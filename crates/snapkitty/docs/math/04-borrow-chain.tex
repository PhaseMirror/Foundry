\section{Linear Type Theory}
A resource $r$ has type $A$ with \emph{multiplicity} $m \in \{0, 1, \omega\}$:
\begin{itemize}
  \item $m = 0$: erased (never used)
  \item $m = 1$: \textbf{linear} (used exactly once --- \texttt{GHC LinearTypes})
  \item $m = \omega$: unrestricted
\end{itemize}

The typing rule for a linear term $e : A\, \%1$:
\begin{equation}
  \frac{\Gamma \vdash e : A\, \%1}{\Gamma \text{ must consume } e \text{ exactly once}}
\end{equation}

\subsection{No-Cloning Theorem (Haskell)}
\texttt{QuantumPipelineState} carries \texttt{QuantumTemp \%1} in its \texttt{Superposed} constructor:
\begin{equation}
  \texttt{Superposed} :: \texttt{QuantumTemp \%1} \to \texttt{QuantumPipelineState}

\section{Verdict Algebra}
Priority ordering $\prec$ over the verdict set $V = \{\texttt{approve}, \texttt{defer}, \texttt{reject}, \texttt{human\_required}, \texttt{escalate}\}$:
\begin{equation}
  \texttt{approve} \prec \texttt{defer} \prec \texttt{reject} \prec \texttt{human\_required} \prec \texttt{escalate}
\end{equation}
Numerically: $p(\texttt{approve}) = 0,\; p(\texttt{defer}) = 1,\; p(\texttt{reject}) = 2,\; p(\texttt{human\_required}) = 3,\; p(\texttt{escalate}) = 4$.

\subsection{Combination}
For a list of verdicts $v_1, \ldots, v_n$:
\begin{equation}
  \operatorname{combine}(v_1, \ldots, v_n) = \underset{i}{\arg\max_\prec}\; \{v_i\}
\end{equation}
Special case: multiple \texttt{human\_required} verdicts \emph{merge} their policy ID lists (not overwrite).
