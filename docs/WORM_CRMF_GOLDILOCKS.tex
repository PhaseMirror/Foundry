    worm_event_t* events;
    size_t count;
    size_t capacity;
    uint8_t last_hash[32];
} worm_log_t;

void worm_log_event(worm_log_t* log, const char* type, 
                    uint32_t src, uint32_t tgt, const char* reason) {
    worm_event_t event = {0};
    strcpy(event.event_type, type);
    event.src_prime = src;
    event.tgt_prime = tgt;
    strcpy(event.reason, reason);
    event.timestamp = time(NULL);
    event.snapshot = hydration_fsm_capture();
    
    // Compute hash chain
    sha256(&event, sizeof(event), event.event_hash);
    memcpy(event.prev_hash, log->last_hash, 32);
    
    // Recompute hash with prev_hash
    uint8_t chained[256];
    memcpy(chained, &event, sizeof(event));
    sha256(chained, sizeof(chained), event.event_hash);
    
    log->events[log->count++] = event;
    memcpy(log->last_hash, event.event_hash, 32);
}
