## 2. Production‑Ready `build.rs` (for `crates/sigma/build.rs`)

This script:
- Writes generated Rust structs directly into **`OUT_DIR`**.
- Uses **tight `rerun-if-changed`** watches (only the Lean files that affect the generation).
- Supports the **`lean-gen`** feature flag (enabled by default, disabled with `--no-default-features` or `--features` toggles).
- Skips the **C++ generation** per our decision to postpone it (non‑fatal, left as a commented placeholder).
- Handles all errors explicitly and fails the build cleanly if something goes wrong.

```rust
// crates/sigma/build.rs
use std::env;
use std::fs;
use std::path::PathBuf;
use std::process::Command;

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let out_dir = PathBuf::from(env::var("OUT_DIR")?);
    let lean_root = PathBuf::from("../lean");

    // --- Watch only the Lean sources that affect the generated structs ---
    // This prevents spurious rebuilds when unrelated Lean files change.
    println!("cargo:rerun-if-changed={}", lean_root.join("lakefile.lean").display());
    println!("cargo:rerun-if-changed={}", lean_root.join("ExportThresholds.lean").display());
    // If ExportThresholds.lean imports other Lean files, add them here too.
    // e.g., println!("cargo:rerun-if-changed={}", lean_root.join("MyInvariants.lean").display());

    // --- Feature: lean-gen (enabled by default) ---
    #[cfg(feature = "lean-gen")]
    {
        println!("cargo:info=Lean generation enabled – running Lean Meta...");

        // Ensure OUT_DIR exists.
        fs::create_dir_all(&out_dir)?;

        // Invoke the Lean Meta generator.
        // We pass the output directory as a CLI argument so the Lean script can
        // write directly to OUT_DIR, avoiding a copy step.
        let status = Command::new("lake")
            .args([
                "run",
                "export_thresholds",
                "--",
                "--out-dir",
                &out_dir.to_string_lossy(),
            ])
            .current_dir(&lean_root)
            .status()?;

        if !status.success() {
            return Err(
                "Lean Meta generation (export_thresholds) failed.\n\
                 Ensure Lean 4 and `lake` are installed, and the Lean project builds.\n\
                 Hint: run `lake build` in ../lean to test manually."
                    .into(),
            );
        }

        println!("cargo:info=✅ Lean generation succeeded – written to {}", out_dir.display());
    }

    // --- Fallback: feature disabled -> use checked‑in version ---
    #[cfg(not(feature = "lean-gen"))]
    {
        let fallback_path = PathBuf::from("src/generated/thresholds.rs");
        if !fallback_path.exists() {
            return Err(
                "Feature 'lean-gen' is disabled, but the fallback file \
                 `src/generated/thresholds.rs` is missing.\n\
                 Either check it into the repository or enable the 'lean-gen' feature."
                    .into(),
            );
        }

        fs::create_dir_all(&out_dir)?;
        fs::copy(&fallback_path, out_dir.join("thresholds.rs"))?;
        println!("cargo:info=Using checked‑in generated thresholds.rs (lean-gen disabled).");
    }

    // --- Optional C++ generation (postponed – kept as a placeholder) ---
    // If we later need a C++ header, we can uncomment this block and gate it
    // behind a `cpp-gen` feature. For now, we skip it entirely.
    //
    // #[cfg(feature = "cpp-gen")] {
    //     let cpp_status = Command::new("lake")
    //         .args(["run", "generate_cpp_ffi", "--", "--out-dir", &out_dir.to_string_lossy()])
    //         .current_dir(&lean_root)
    //         .status()?;
    //     if !cpp_status.success() {
    //         eprintln!("cargo:warning=Lean → C++ FFI generation failed (non‑fatal)");
    //     }
    // }

    Ok(())
}
```

import Lean.Meta
import System.IO

def generate_rust_struct (out_dir : String) : IO Unit := do
  let rust_code := s!"#[derive(Debug, Clone, serde::Serialize, serde::Deserialize, postcard::Serialize, postcard::Deserialize)]
pub struct Thresholds {{
    pub tau_r: f64,
    pub l_eff_max: f64,

    pub atlas_signature: (i32, i32),
    pub r_sc_reference: f64,
    pub contractivity_margin: f64,

  let outDir := match args with
    | ["--out-dir", dir] => dir
    | _ => "generated"  -- fallback for manual testing
  generate_rust_struct outDir
