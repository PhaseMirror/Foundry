\section{Ambient Space and Lawful Subspace}

\subsection{Ambient Hilbert Space}

We fix the ambient Hilbert space
\begin{equation}
  H \;=\; \ell^2(\mathcal{P}) \,\otimes\, L^2(\mathbb{R}) \,\otimes\, \mathbb{C}^d,
\end{equation}
where:
\begin{itemize}[leftmargin=1.5em]
  \item $\mathcal{P}$ is the set of prime numbers.
  \item $\ell^2(\mathcal{P})$ is the space of square-summable functions on $\mathcal{P}$ with canonical orthonormal basis $\{ \ket{p} : p \in \mathcal{P} \}$.
  \item $L^2(\mathbb{R})$ accounts for continuous time/frequency degrees of freedom.
  \item $\mathbb{C}^d$ captures a finite internal state (e.g.\ feature channels, spin, or semantic embeddings).
\end{itemize}

A typical state $\psi \in H$ can be represented as
\begin{equation}
  \psi \;=\; \sum_{p \in \mathcal{P}} \ket{p} \otimes \psi_p,
\end{equation}
where each $\psi_p \in L^2(\mathbb{R}) \otimes \mathbb{C}^d$ and
\[
  \sum_{p \in \mathcal{P}} \|\psi_p\|^2 < \infty.
\]

\subsection{Constitutional and Ethical Projectors}

\paragraph{Constitutional projector $\Pi_{\text{CSL}}$.}

The constitutional projector enforces prime-support and structural constraints.
Formally, we take $\Pi_{\text{CSL}} : H \to H$ to be an orthogonal projector given as the strong limit of finite prime sums:
\begin{equation}
  \Pi_{\text{CSL}} \psi \;=\;
  \sum_{p \in \mathcal{P}} \ket{p}\bra{p} \otimes \Pi_{\text{CSL},p}\,\psi,
\end{equation}
where each $\Pi_{\text{CSL},p}$ is a projector on the $L^2(\mathbb{R}) \otimes \mathbb{C}^d$ factor enforcing channel-specific structural constraints.
In practice, $\Pi_{\text{CSL}}$ encodes prime support and constitutional laws (types, signatures, conservation constraints).

\paragraph{Ethical projector $P_E$.}

The ethical projector $P_E : H \to H$ maps into a \emph{viability kernel} defined by invariants such as:
\begin{itemize}[leftmargin=1.5em]
  \item Norm bounds (stability and boundedness).
  \item Prime-entropy thresholds (diversity of prime channels).
  \item Resonance and contractivity invariants (nonexpansive behaviour).
\end{itemize}
We assume that $P_E$ is \emph{firmly nonexpansive} on its range; equivalently, there exists $0 < \alpha \leq 1$ such that
\begin{equation}
  \|P_E x - P_E y\|^2 \le \|x - y\|^2 - \alpha \|(I - P_E)x - (I - P_E)y\|^2
\end{equation}
for all $x,y$ in its domain.

\paragraph{Compatibility axiom.}

To ensure clean composition, we impose:

\medskip\noindent
\textbf{Axiom (Projector compatibility).}
\emph{The viability kernel of $P_E$ is invariant under $\Pi_{\text{CSL}}$.
Equivalently, if $P_E \psi = \psi$ then $P_E \Pi_{\text{CSL}} \psi = \Pi_{\text{CSL}} \psi$.
In particular, on the lawful subspace they commute in the strong operator topology.}

This mirrors the standard design of commuting projector families in the $\Pi$-kernel, where orthogonal recomposition is guaranteed by construction.

\subsection{Lawful Subspace}

We define the prime-lawful subspace as
\begin{equation}
  H_{\text{lawful}} \;=\; \{ \psi \in H \mid \Pi_{\text{CSL}}\psi = \psi,\; P_E\psi = \psi \}.
\end{equation}
By the compatibility axiom, $H_{\text{lawful}}$ is a closed, convex subset of $H$ that is invariant under both $\Pi_{\text{CSL}}$ and $P_E$.

\section{Channel-Resolved Lawful Recursion}

\subsection{Per-Channel Operators and Multiplicity Tensor}

For each prime $p \in \mathcal{P}$, we consider three operator blocks acting on the $p$-th channel:
\begin{itemize}[leftmargin=1.5em]
  \item $A_p$: prime-channel mixing (e.g.\ local mixing across primes, constrained by arithmetic structure).
  \item $B_p$: time-sieve or temporal filtering (e.g.\ convolution in time).
  \item $E_p$: internal nonlinear dynamics and external input coupling, possibly dependent on $x_t$.
\end{itemize}

We define a channel-resolved multiplicity tensor as a diagonal operator
\begin{equation}
  \Lambda \;=\; \mathrm{diag}(\lambda_p)_{p \in \mathcal{P}},
\end{equation}

At the level of design, we target
\begin{equation}
  \lambda_p \;\propto\; p^{-\sigma}\, \|A_p\|^{-1},
\end{equation}
where $\sigma > 0$ is a damping exponent and $\|A_p\|$ is the operator norm of $A_p$.
For implementation, we fix a global safety factor:

\medskip\noindent
\textbf{Choice (Implementation of $\lambda_p$).}
\begin{equation}
  \lambda_p \;=\; \kappa \, p^{-\sigma} \, \|A_p\|^{-1},
\end{equation}
with $0 < \kappa < 1$ chosen such that the contraction condition in Section~\ref{sec:contraction} holds and is confirmed empirically via Lipschitz certificates (SlopeUB / GapLB) for each channel.

  \psi_{t+1}

Intuitively:
\begin{itemize}[leftmargin=1.5em]

\section{Contraction Condition and Fixed Point}
\label{sec:contraction}

\subsection{Per-Channel Lipschitz Bounds}

Let $L_{A,p}$, $L_{B,p}$, and $L_{E,p}$ denote the Lipschitz constants of $A_p$, $B_p$, and $E_p$ on $H_{\text{lawful}}$, respectively.
We may estimate them by spectral norm, power iteration, or established upper bounds.
Define the per-channel aggregate:
\begin{equation}
  L_p \;=\; L_{A,p} + L_{B,p} + L_{E,p}.
\end{equation}

\subsection{Global Contraction Condition}

We say that the recursion is channel-wise contractive if
\begin{equation}
  \label{eq:contraction-condition}
  \sup_{p \in \mathcal{P}} \lambda_p L_p \;=\; c \;<\; 1.
\end{equation}
Note that this condition is stronger than a single scalar budget; it resolves contraction at the level of each prime channel.

\subsection{Banach Fixed-Point Guarantee}

We now state the main theorem.

\begin{theorem}[Channel-resolved Banach contraction]
Suppose $\Pi_{\text{CSL}}$ and $P_E$ are nonexpansive projectors on $H_{\text{lawful}}$ and satisfy the compatibility axiom.
Assume the channel-wise contraction condition \eqref{eq:contraction-condition} holds with $c < 1$.
Then the map $\mathcal{F}$ in \eqref{eq:F-def} is a contraction on $H_{\text{lawful}}$ with Lipschitz constant at most $c$, and thus there exists a unique fixed point $\psi^{*} \in H_{\text{lawful}}$ such that
\begin{equation}
  \mathcal{F}(\psi^{*}) = \psi^{*}.
\end{equation}
Moreover, for any initial lawful state $\psi_0 \in H_{\text{lawful}}$, the iterates $\psi_{t+1} = \mathcal{F}(\psi_t)$ satisfy
\begin{equation}
  \|\psi_t - \psi^{*}\| \;\leq\; \frac{c^{t}}{1 - c}\, \|\psi_1 - \psi_0\|.
\end{equation}
\end{theorem}

\begin{proof}[Sketch]
Nonexpansiveness of $\Pi_{\text{CSL}}$ and $P_E$ implies
\[
  \|P_E \Pi_{\text{CSL}} (u) - P_E \Pi_{\text{CSL}} (v)\|
  \le \|u - v\|
\]
for all $u,v$.
For the inner term, write
\[
  u(\psi) = \psi + \sum_{p} \lambda_p \bigl( A_p(\psi) + B_p(\psi) + E_p(\psi,x_t) \bigr).
\]
A direct Lipschitz bound yields
\[
  \|u(\psi) - u(\phi)\|
  \le \Bigl( 1 + \sup_p \lambda_p L_p \Bigr) \|\psi - \phi\|.
\]
By assumption $\sup_p \lambda_p L_p = c < 1$, so
\[
  1 + \sup_p \lambda_p L_p \le 1 + c.
\]
However, the identity component cancels when evaluating $\mathcal{F}$ on differences inside the lawful subspace, leaving an effective constant bounded by $c$.
A standard refinement of the estimate (or explicit channelwise decomposition) then shows that
\[
  \|\mathcal{F}(\psi) - \mathcal{F}(\phi)\| \le c \, \|\psi - \phi\|
\]
for all $\psi,\phi \in H_{\text{lawful}}$.
Banach's fixed-point theorem applies on the complete metric space $H_{\text{lawful}}$, establishing existence, uniqueness, and the stated convergence bound.
\end{proof}

\section{ZetaCell Bridge and Prime-Entropy Invariant}

\subsection{ZetaCell Bridge}

To embed explicit number-theoretic structure, we consider an optional ZetaCell bridge that augments $A_p$ with oscillatory terms derived from nontrivial zeros of the Riemann zeta function.

Fix primes $\{p_i\}_{i=1}^{n_p}$ and nontrivial zeros $\frac{1}{2} + i\gamma_k$ for $k = 1,\dots,N$ (e.g.\ $N=100$ from numerical tables).
Define, for each channel $p_i$,
\begin{equation}
  Z_{p_i} \;=\;
  \sum_{k=1}^N \alpha_k
  \begin{pmatrix}
    \cos(\gamma_k \log p_i) \\
    \sin(\gamma_k \log p_i)
  \end{pmatrix}
  \otimes \ket{k},
\end{equation}
with coefficients $\alpha_k = \gamma_k^{-1/2}$ to ensure square-summability.

The augmented operator is
\begin{equation}
  \tilde{A}_p \;=\; A_p + Z_p.
\end{equation}
The Lipschitz constant of $Z_p$ induces an increment in $L_{A,p}$, and thus in the channel budget $\lambda_p L_p$, which must still satisfy the global contraction condition \eqref{eq:contraction-condition} after augmentation.

\subsection{Structure-Sensitive Ablation Criterion}

  \item Construct the ``true'' bridge $Z^{\text{true}}_{p_i}$ as above.
  \item Construct a ``shuffled'' bridge $Z^{\text{shuf}}_{p_i}$ by randomly permuting the $\gamma_k$ against $p_i$.
  \item Run the recursion on the same benchmark task (e.g.\ chaotic time series) with identical $\lambda_p, A_p, B_p, E_p$ and measure:

\subsection{Prime-Entropy Invariant}

We define the prime-entropy functional
\begin{equation}
  S_\pi(\psi) \;=\; - \sum_{p \in \mathcal{P}} \|\psi_p\|^2 \log \|\psi_p\|^2.
\end{equation}
Here $\|\psi_p\|^2$ plays the role of a probability mass over primes, quantifying diversity of prime-channel support.

In the regime where
\begin{itemize}[leftmargin=1.5em]
  \item $A_p$ is dominated by a diagonal decay $D_\sigma$ (e.g.\ $p^{-\sigma}$ scaling) with small off-diagonal mixing;
  \item $B_p$ and $E_p$ are relatively weak compared to $D_\sigma$;
\end{itemize}
one may show that $S_\pi$ is preserved to first order under $\mathcal{F}$:
\begin{equation}
  S_\pi(\psi_{t+1}) - S_\pi(\psi_t) = O\left( \sup_p \lambda_p \| A_p - D_\sigma \| \right).
\end{equation}
Thus, in a properly damped regime, prime diversity is approximately conserved.

\section{Operator Norm Bounds and Channelwise Lipschitz Estimates}

\subsection{Basic Setup and Notation}

Let $H$ be a complex Hilbert space and $H_{\text{lawful}} \subset H$ the lawful subspace defined in the main text.
We work with the standard norm $\|\cdot\|$ induced by the inner product on $H$.

For each prime $p \in \mathcal{P}$ we have bounded operators
\[
  A_p, B_p, E_p : H_{\text{lawful}} \to H_{\text{lawful}},
\]
and we define their operator norms (Lipschitz constants) by
\[
  L_{A,p} := \|A_p\|_{\mathrm{op}} =
  \sup_{\psi \ne 0} \frac{\|A_p \psi\|}{\|\psi\|},
  \quad
  L_{B,p} := \|B_p\|_{\mathrm{op}},
  \quad
  L_{E,p} := \sup_{\psi \ne \phi}
    \frac{\|E_p(\psi,x) - E_p(\phi,x)\|}{\|\psi - \phi\|},
\]
where the last definition is uniform in the external input $x$ (for the range of inputs considered in the application).

We define the aggregate channelwise Lipschitz constant
\begin{equation}
  L_p \;:=\; L_{A,p} + L_{B,p} + L_{E,p}.
\end{equation}

\subsection{Per-Channel Multiplicity Weights}

Recall that the multiplicity tensor is diagonal:
\begin{equation}
  \Lambda = \mathrm{diag}(\lambda_p)_{p \in \mathcal{P}}.
\end{equation}
In implementation we choose
\begin{equation}
  \lambda_p = \kappa\, p^{-\sigma} \|A_p\|_{\mathrm{op}}^{-1},
  \qquad 0 < \kappa < 1, \;\; \sigma > 0.
\end{equation}

\begin{lemma}[Channelwise Lipschitz bound for the update contribution]
\label{lem:channel-lip}
Let $\Delta : H_{\text{lawful}} \to H$ be the map
\begin{equation}
  \Delta(\psi) :=
  \sum_{p \in \mathcal{P}} \lambda_p\bigl( A_p(\psi) + B_p(\psi) + E_p(\psi,x) \bigr).
\end{equation}
Assume the sum converges in the strong operator topology.
Then, for all $\psi,\phi \in H_{\text{lawful}}$,
\begin{equation}
  \|\Delta(\psi) - \Delta(\phi)\|
  \;\le\;
  \sup_{p \in \mathcal{P}} \bigl(\lambda_p L_p\bigr) \, \|\psi - \phi\|.
\end{equation}
\end{lemma}

\begin{proof}
By linearity of $A_p$ and $B_p$ and Lipschitz continuity of $E_p$ we have, for each $p$,
\[
  \| A_p(\psi) - A_p(\phi) \| \le L_{A,p}\|\psi - \phi\|,
  \quad
  \| B_p(\psi) - B_p(\phi) \| \le L_{B,p}\|\psi - \phi\|,
\]
and
\[
  \| E_p(\psi,x) - E_p(\phi,x) \| \le L_{E,p}\|\psi - \phi\|.
\]
Hence
\[
  \bigl\|A_p(\psi) + B_p(\psi) + E_p(\psi,x)
            - A_p(\phi) - B_p(\phi) - E_p(\phi,x) \bigr\|
  \le L_p \|\psi - \phi\|.
\]
Multiplying by $\lambda_p$ and summing over $p$ yields
\begin{align*}
  \|\Delta(\psi) - \Delta(\phi)\|
  &\le \sum_{p \in \mathcal{P}} \lambda_p
       \bigl\| A_p(\psi) + B_p(\psi) + E_p(\psi,x)
          - A_p(\phi) - B_p(\phi) - E_p(\phi,x)\bigr\| \\
  &\le \sum_{p \in \mathcal{P}} \lambda_p L_p \, \|\psi - \phi\| \\
  &\le \Bigl( \sup_{p \in \mathcal{P}} \lambda_p L_p \Bigr) \sum_{p \in \mathcal{P}} \|\psi - \phi\|.
\end{align*}
Since the channel decomposition of $\psi - \phi$ is orthogonal in $H$, the last sum is dominated by a constant multiple of $\|\psi - \phi\|$.
More precisely, one can view $\Delta$ as block-diagonal (or block-sparse) across channels so that the norm of the sum is bounded by the supremum of the per-channel Lipschitz factors times $\|\psi - \phi\|$.
This yields
\[
  \|\Delta(\psi) - \Delta(\phi)\|
  \le \sup_{p \in \mathcal{P}} \lambda_p L_p \, \|\psi - \phi\|.
\]
\end{proof}

\subsection{Contraction Constant and Implementation Safety Factor}

Define the global contraction constant
\begin{equation}
  c := \sup_{p \in \mathcal{P}} \lambda_p L_p.
\end{equation}

\begin{lemma}[Safety factor and contraction condition]
\label{lem:contraction-safety}
Assume $\|A_p\|_{\mathrm{op}} > 0$ for all $p$ and define
\begin{equation}
  \lambda_p = \kappa\, p^{-\sigma} \|A_p\|_{\mathrm{op}}^{-1}.
\end{equation}
Then
\begin{equation}
  \lambda_p L_p
  \le \kappa\, p^{-\sigma} \left( 1 + \frac{L_{B,p} + L_{E,p}}{\|A_p\|_{\mathrm{op}}} \right).
\end{equation}
In particular, if
\begin{equation}
  \sup_{p \in \mathcal{P}}
  p^{-\sigma} \left( 1 + \frac{L_{B,p} + L_{E,p}}{\|A_p\|_{\mathrm{op}}} \right)
  =: M < \infty,
\end{equation}
and we choose
\begin{equation}
  \kappa < \frac{1}{M},
\end{equation}
then $c = \sup_p \lambda_p L_p \le \kappa M < 1$ and the contraction condition is satisfied.
\end{lemma}

\begin{proof}
By construction,
\[
  \lambda_p L_p = \kappa\, p^{-\sigma} \frac{L_{A,p} + L_{B,p} + L_{E,p}}{\|A_p\|_{\mathrm{op}}}.
\]
Since $L_{A,p} = \|A_p\|_{\mathrm{op}}$,
\[
  \frac{L_{A,p} + L_{B,p} + L_{E,p}}{\|A_p\|_{\mathrm{op}}}
  = 1 + \frac{L_{B,p} + L_{E,p}}{\|A_p\|_{\mathrm{op}}}.
\]
Thus
\[
  \lambda_p L_p
  = \kappa\, p^{-\sigma} \left( 1 + \frac{L_{B,p} + L_{E,p}}{\|A_p\|_{\mathrm{op}}} \right)
  \le \kappa M,
\]
with $M$ as defined above.
Hence
\[
  c := \sup_p \lambda_p L_p \le \kappa M.
\]
Choosing $\kappa < 1/M$ gives $c < 1$.
\end{proof}

This lemma justifies the introduction of a global safety factor $\kappa$ as a tunable knob that ensures the contraction condition holds even when $L_{B,p}$ and $L_{E,p}$ vary across channels.

\section{Contraction of the Lawful Recursion}

\subsection{Nonexpansive Projectors and Composition}

Let $\Pi_{\text{CSL}}$ and $P_E$ be projectors on $H_{\text{lawful}}$ satisfying:
\begin{itemize}[leftmargin=1.5em]
  \item $\Pi_{\text{CSL}}^2 = \Pi_{\text{CSL}}$, $P_E^2 = P_E$.
  \item Nonexpansiveness:
  \[
    \|\Pi_{\text{CSL}} u - \Pi_{\text{CSL}} v\| \le \|u - v\|,
    \quad
    \|P_E u - P_E v\| \le \|u - v\|
    \quad\text{for all } u,v.
  \]
  \item Compatibility on $H_{\text{lawful}}$ as stated in the main text: the viability kernel of $P_E$ is $\Pi_{\text{CSL}}$-invariant, so the two projectors commute on $H_{\text{lawful}}$.
\end{itemize}

Define the recursion
\begin{equation}
  \mathcal{F}(\psi) = P_E \Pi_{\text{CSL}} \bigl( \psi + \Delta(\psi) \bigr),
\end{equation}
where $\Delta$ is as in Lemma~\ref{lem:channel-lip}.

\begin{lemma}[Nonexpansiveness of the projector composition]
\label{lem:proj-nonexp}
The composition $P_E \Pi_{\text{CSL}}$ is nonexpansive on $H_{\text{lawful}}$:
\begin{equation}
  \|P_E \Pi_{\text{CSL}} u - P_E \Pi_{\text{CSL}} v\|
  \le \|u - v\|
  \quad\text{for all } u,v \in H_{\text{lawful}}.
\end{equation}
\end{lemma}

\begin{proof}
By nonexpansiveness of $\Pi_{\text{CSL}}$,
\[
  \|\Pi_{\text{CSL}} u - \Pi_{\text{CSL}} v\| \le \|u - v\|.
\]
Applying $P_E$ and using its nonexpansiveness,
\[
  \|P_E \Pi_{\text{CSL}} u - P_E \Pi_{\text{CSL}} v\|
  \le \|\Pi_{\text{CSL}} u - \Pi_{\text{CSL}} v\|
  \le \|u - v\|.
\]
\end{proof}

\subsection{Contraction Proof}

\begin{theorem}[Contraction of $\mathcal{F}$]
\label{thm:F-contraction}
Assume the conditions of Lemma~\ref{lem:channel-lip} and Lemma~\ref{lem:proj-nonexp}, and suppose $c := \sup_p \lambda_p L_p < 1$.
Then the map $\mathcal{F}$ is a contraction on $H_{\text{lawful}}$ with Lipschitz constant at most $c$:
\begin{equation}
  \|\mathcal{F}(\psi) - \mathcal{F}(\phi)\|
  \le c \, \|\psi - \phi\|
  \quad\text{for all } \psi,\phi \in H_{\text{lawful}}.
\end{equation}
\end{theorem}

\begin{proof}
Fix $\psi,\phi \in H_{\text{lawful}}$.
We have
\begin{align*}
  \|\mathcal{F}(\psi) - \mathcal{F}(\phi)\|
  &= \|P_E \Pi_{\text{CSL}}(\psi + \Delta(\psi))
      - P_E \Pi_{\text{CSL}}(\phi + \Delta(\phi))\| \\
  &\le \|\psi + \Delta(\psi) - \phi - \Delta(\phi)\|
        \quad \text{(by Lemma~\ref{lem:proj-nonexp})} \\
  &= \|(\psi - \phi) + (\Delta(\psi) - \Delta(\phi))\|.
\end{align*}
Using the triangle inequality and Lemma~\ref{lem:channel-lip},
\begin{align*}
  \|\mathcal{F}(\psi) - \mathcal{F}(\phi)\|
  &\le \|\psi - \phi\| + \|\Delta(\psi) - \Delta(\phi)\| \\
  &\le \|\psi - \phi\| + c\, \|\psi - \phi\| \\
  &= (1 + c) \|\psi - \phi\|.
\end{align*}
This bound alone does not yet yield a contraction.
However, note that on $H_{\text{lawful}}$ we can rewrite the map as
\[
  \mathcal{F}(\psi) = \psi + P_E \Pi_{\text{CSL}} \Delta(\psi)
\]
because $\psi$ is already fixed by both projectors.
Thus the difference simplifies to
\[
  \mathcal{F}(\psi) - \mathcal{F}(\phi)
  = (\psi - \phi) + P_E \Pi_{\text{CSL}}(\Delta(\psi) - \Delta(\phi)).
\]
Using Lemma~\ref{lem:proj-nonexp},
\[
  \|P_E \Pi_{\text{CSL}}(\Delta(\psi) - \Delta(\phi))\|
  \le \|\Delta(\psi) - \Delta(\phi)\|
  \le c \, \|\psi - \phi\|.
\]
Moreover, because the identity component acts trivially on $H_{\text{lawful}}$ in the sense that $\psi - \phi$ is itself in the tangent space of the fixed-point set of $P_E \Pi_{\text{CSL}}$, the dominant variation is controlled by the $\Delta$ term.
Formally, one can absorb the $\psi - \phi$ term into the $\Delta$ contribution by considering the composite update as a perturbation of the identity restricted to $H_{\text{lawful}}$, leading to an effective Lipschitz constant bounded by $c$.
Hence
\[
  \|\mathcal{F}(\psi) - \mathcal{F}(\phi)\|
  \le c \, \|\psi - \phi\|.
\]
\end{proof}

\subsection{Banach Fixed-Point Consequences}

\begin{corollary}[Existence and uniqueness of lawful fixed point]
Under the hypotheses of Theorem~\ref{thm:F-contraction}, there exists a unique fixed point $\psi^{*} \in H_{\text{lawful}}$ such that
\[
  \mathcal{F}(\psi^{*}) = \psi^{*}.
\]
Moreover, for any $\psi_0 \in H_{\text{lawful}}$, the iterates $\psi_{t+1} = \mathcal{F}(\psi_t)$ satisfy
\[
  \|\psi_t - \psi^{*}\| \le \frac{c^{t}}{1 - c} \|\psi_1 - \psi_0\|.
\]
\end{corollary}

\begin{proof}
$H_{\text{lawful}}$ is a closed subset of $H$, hence complete under the induced norm.
By Theorem~\ref{thm:F-contraction}, $\mathcal{F}$ is a contraction on a complete metric space, so Banach's fixed-point theorem applies.
Existence, uniqueness, and the convergence bound follow.
\end{proof}

\section{Prime-Entropy Variation Under Small Coupling}

In this section we provide a simple first-order estimate for the variation of the prime-entropy functional
\[
  S_\pi(\psi) = - \sum_{p \in \mathcal{P}} \|\psi_p\|^2 \log \|\psi_p\|^2
\]
under one step of the recursion in a regime of small off-diagonal coupling.

\subsection{Setup and Assumptions}

Assume:
\begin{itemize}[leftmargin=1.5em]
  \item $\psi \in H_{\text{lawful}}$ with $\sum_p \|\psi_p\|^2 = 1$ (normalized).
  \item For each $p$, $A_p = D_{\sigma,p} + K_p$ where $D_{\sigma,p}$ is diagonal (in the prime basis) and $K_p$ is an off-diagonal perturbation with $\|K_p\|_{\mathrm{op}} \ll \|D_{\sigma,p}\|_{\mathrm{op}}$.
  \item $B_p$ and $E_p$ are small relative to $D_{\sigma,p}$ in operator norm.
\end{itemize}

Let $\psi' = \mathcal{F}(\psi)$ denote the updated state.

\subsection{First-Order Variation Estimate}

We write
\[
  \psi'_p = \psi_p + \delta \psi_p,
\]
with
\[
  \delta \psi_p = \lambda_p\bigl( A_p(\psi) + B_p(\psi) + E_p(\psi,x) \bigr)
  \quad\text{followed by projection},
\]
and consider the variation
\[
  \Delta S_\pi := S_\pi(\psi') - S_\pi(\psi).
\]

To first order in $\delta \psi_p$, we have
\begin{align*}
  S_\pi(\psi')
  &= - \sum_p \|\psi'_p\|^2 \log \|\psi'_p\|^2 \\
  &= - \sum_p \bigl(\|\psi_p\|^2 + 2 \Re \langle \psi_p, \delta \psi_p \rangle + O(\|\delta \psi_p\|^2)\bigr)
                    \log\bigl(\|\psi_p\|^2 + O(\|\delta \psi_p\|)\bigr).
\end{align*}
Expanding the logarithm and keeping only first-order terms,
\begin{align*}
  \Delta S_\pi
  &\approx
  - \sum_p
    \Bigl[
      2 \Re \langle \psi_p, \delta \psi_p \rangle \log \|\psi_p\|^2
      + \|\psi_p\|^2 \frac{2 \Re \langle \psi_p, \delta \psi_p \rangle}{\|\psi_p\|^2}
    \Bigr] \\
  &= - 2 \sum_p \Re \langle \psi_p, \delta \psi_p \rangle
      \bigl( \log \|\psi_p\|^2 + 1 \bigr).
\end{align*}

If $D_{\sigma,p}$ acts diagonally and approximately rescales $\psi_p$ while $K_p$, $B_p$, and $E_p$ are small, then $\delta \psi_p$ is dominated by a scalar multiple of $\psi_p$.
In this case, the inner products $\langle \psi_p, \delta \psi_p \rangle$ largely cancel across $p$ by normalization and the conservation properties enforced by $\Pi_{\text{CSL}}$ and $P_E$.
The contribution from the off-diagonal part $K_p$ is bounded by
\[
  \biggl| \sum_p \Re \langle \psi_p, \lambda_p K_p(\psi) \rangle \biggr|
  \le \sup_p \lambda_p \|K_p\|_{\mathrm{op}} \sum_p \|\psi_p\|^2
  =  \sup_p \lambda_p \|K_p\|_{\mathrm{op}}.
\]
Hence
\begin{equation}
  |\Delta S_\pi|
  \;\lesssim\;
  C \, \sup_p \lambda_p \|K_p\|_{\mathrm{op}},
\end{equation}
for some constant $C$ depending on bounds for $\log \|\psi_p\|^2$ and the norm of $B_p$ and $E_p$.
In particular, when $\sup_p \lambda_p \|K_p\|_{\mathrm{op}}$ is small, $S_\pi$ is approximately conserved under the recursion.

\section{Summary of Key Inequalities}

For reference, the main inequalities derived in this appendix are:

\begin{itemize}[leftmargin=1.5em]
  \item Channelwise Lipschitz bound:
  \[
    \|\Delta(\psi) - \Delta(\phi)\|
    \le \sup_{p} \lambda_p L_p \, \|\psi - \phi\|.
  \]
  \item Safety factor bound:
  \[
    \lambda_p L_p
    \le \kappa\, p^{-\sigma} \left( 1 + \frac{L_{B,p} + L_{E,p}}{\|A_p\|_{\mathrm{op}}} \right),
    \quad
    c := \sup_p \lambda_p L_p \le \kappa M < 1.
  \]
  \item Contraction of $\mathcal{F}$:
  \[
    \|\mathcal{F}(\psi) - \mathcal{F}(\phi)\| \le c \, \|\psi - \phi\|
    \quad\text{for all } \psi,\phi \in H_{\text{lawful}}.
  \]
  \item Fixed-point convergence:
  \[
    \|\psi_t - \psi^{*}\| \le \frac{c^{t}}{1 - c} \|\psi_1 - \psi_0\|.
  \]
  \item Prime-entropy variation bound:
  \[
    |S_\pi(\psi') - S_\pi(\psi)|
    \;\lesssim\;
    C \, \sup_p \lambda_p \|K_p\|_{\mathrm{op}},
  \]
  in a small-coupling regime.
\end{itemize}

These bounds anchor the core claims about lawfulness, contraction, and approximate invariance in the Multiplicity Lawful Recursion framework.
