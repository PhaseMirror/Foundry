\section*{Primitives}
Let the four primitives be real numbers scaled to $[0,1]$:
\[
F \text{ (Focus/Clarity, }\triangle),\quad
C \text{ (Connection, }\circ),\quad
G \text{ (Growth/Memory-in-motion, }\spiral),\quad
S \text{ (Structure/Stability, }\square).
\]

\section*{Canonical weights and proportional core}
Use the $1{:}2{:}4{:}6$ recipe (``12-node'' proportions). Define
\[
\mathbf w=\Bigl(\tfrac{1}{13},\,\tfrac{2}{13},\,\tfrac{4}{13},\,\tfrac{6}{13}\Bigr),
\qquad
P \;=\; \mathbf w\!\cdot\!(F,C,G,S)
\;=\; \frac{1F + 2C + 4G + 6S}{13}\in[0,1].
\]

\section*{Geometric coherence (12-axis phase order parameter)}
Let $\{\phi_n\}_{n=1}^{12}$ be twelve salient axes/angles, e.g. drawn from
$\{36^\circ,72^\circ,108^\circ,\ldots\}$. Define the unit-circle phase coherence
\[
\kappa_{\mathrm{geom}}
\;=\;
\left|\frac{1}{12}\sum_{n=1}^{12} e^{\,i\phi_n}\right|
\in[0,1],
\quad
\kappa_{\mathrm{geom}}=1 \text{ iff perfect angular alignment.}
\]

\section*{Harmonic resonance (acoustic fit)}
Let $f$ be the dominant acoustic frequency, $v$ the wave speed, and
$\{L_k\}_{k=1}^K$ characteristic lengths. For each $k$ define the mismatch
\[
\delta_k
\;=\;
\min_{n\in\mathbb N}
\left|\frac{f}{\,n\,v/(2L_k)}-1\right|,
\qquad
\overline{\delta} \;=\; \frac{1}{K}\sum_{k=1}^{K}\delta_k.
\]
Use a soft resonance score
\[
R \;=\; \exp\!\bigl(-\alpha\,\overline{\delta}\bigr),
\qquad
\alpha>0,\;\; R\in(0,1],\;\; R=1 \text{ at perfect match.}
\]
\emph{Variant.} If targeting musical or $\varphi$-based ratios, replace
$\,n\,v/(2L_k)$ with the desired ratio target for each $k$.

\section*{Master balance metric}
\[
\boxed{\; T \;=\; P \times \kappa_{\mathrm{geom}} \times R \;}
\qquad
T\in[0,1],\quad
T\to 1 \text{ indicates equilibrium; } T\ll 1 \text{ indicates imbalance.}
\]

\section*{Procedure (quick recipe)}
\begin{enumerate}[leftmargin=*, itemsep=2pt]
  \item Normalize observations: map $(F,C,G,S)\to[0,1]$ from data or ratings.
  \item Compute $P$ via the $1{:}2{:}4{:}6$ weighting.
  \item Measure $\kappa_{\mathrm{geom}}$: extract 12 axes and compute the phase-coherence term.
  \item Estimate $R$: select key lengths $L_k$, compare $f$ to $n\,v/(2L_k)$ (or musical/$\varphi$ targets), then apply $R=\exp(-\alpha\overline{\delta})$.
  \item Report $T$ and track it across changes (design tweaks, materials, pitch, occupancy).
\end{enumerate}

\section*{Notes}
\begin{itemize}[leftmargin=*, itemsep=2pt]
  \item The $1{:}2{:}4{:}6$ weighting is canonical. Projects may learn $\mathbf w^\ast$ but should stay close to preserve the 12-node character.
  \item $\kappa_{\mathrm{geom}}$ is a standard phase-order parameter from synchronization theory.
  \item $R$ is smooth by construction. The exponential avoids hard thresholds and is data-friendly.
  \item Domain-agnostic: architecture, teams, codebases, or cities. Only the measurement maps for $F,C,G,S$ and $(L_k,f)$ change.
\end{itemize}

\section*{One-line summary}
\[
\text{ATLAS--T: Balance emerges when proportions }(P),\; \text{phase/symmetry }(\kappa_{\mathrm{geom}}),\;
\text{and resonance }(R) \text{ coincide.}
\]
