\begin{abstract}
We give a math-first integration of the Automorphic Learning component (finite-group actions, additive masking, unitarization with a Sato--Tate proxy, and ACE projection) with the AU--Math Triad (\emph{Measure}, \emph{Compare}, \emph{Equalize}) used for certified graph learning. The bridge treats attention as a certified graph operator, aligns invariances via permutation-equivariant constructions, and composes certificates (contraction, Lipschitz, KKT/duality) into a single pass/fail gate for safe learning.
\end{abstract}

\section{Objects and Shared State}
\paragraph{Tokens, embeddings, and actions.}
Let $V=\{1,\dots,n\}$ index tokens. Choose odd primes $p_1,p_2$ with $p_1p_2\ge n$ and a CRT injection
$\varphi:V\hookrightarrow \F_{p_1}\times \F_{p_2}$. For an odd prime $p$, set $G_p=\mathrm{AGL}(1,p)=\{i\mapsto u i + k: u\in \F_p^\times,\,k\in\F_p\}$ and act componentwise on $\F_{p_1}\times \F_{p_2}$. Each $g\in G:=G_{p_1}\times G_{p_2}$ induces a permutation matrix $P_g$ on $V$.

\paragraph{Masked additive logits and attention.}
Let $M\in\{0,1\}^{n\times n}$ be a residue mask defined by a Legendre-symbol rule on $\varphi(i)-\varphi(j)$ (non-residue edges banned). With queries/keys $Q,K\in\R^{n\times d}$,
\begin{equation}\label{eq:additive-logits}
L \;=\; \frac{QK^\top}{\sqrt{d}} \;+\; \alpha M\;-\;\beta(\mathbf{1}-M),\qquad \beta\gg 0,\ \alpha\ge 0,
\end{equation}
and the row-stochastic attention $\tilde A=\softmax(L)$.

\paragraph{Sinkhorn unitarization.}
Let $\bB=\mathrm{Sinkhorn}(\mathrm{ReLU}(\tilde A)+\varepsilon)$ be $\varepsilon$-positive and (approximately) bistochastic. Define $\bJ=\frac{1}{n}\one\one^\top$. Two unitarizations:
\begin{align}
\textsf{Exp:}\quad & \bU \;=\; \exp\big(\pi i\, (\bB-\bJ)\big),\\
\textsf{Cayley:}\quad & \bU \;=\; (\Id-\tfrac12(\bB-\bB^\top))(\Id+\tfrac12(\bB-\bB^\top))^{-1}.
\end{align}
Both are permutation-equivariant (see Lem.~\ref{lem:perm-eq}).

\paragraph{Graphization for AU--Triad.}
Symmetrize $\bW=\tfrac12(\bB+\bB^\top)$, degrees $\bD=\diag(\bW\one)$, and Laplacian $\bL=\bD-\bW$. For node signal $u\in\R^n$, the Dirichlet energy is
\begin{equation}\label{eq:dirichlet}
E[u] \;=\; \tfrac12\, u^\top \bL u \;=\; \tfrac12\sum_{i,j} W_{ij}(u_i-u_j)^2.
\end{equation}

\section{Triad Mapping (Measure--Compare--Equalize)}
\subsection*{Measure}
\begin{itemize}[leftmargin=2em]
\item Energies: $E[u]$ in \eqref{eq:dirichlet}; entropy or row-moment summaries of $\tilde A$; unitarity residual $\norm{\bU^*\bU-\Id}_{1\to 1}$.
\item Spectrum: Laplacian eigenvalues $0=\lambda_1\le \lambda_2\le \cdots\le \lambda_n$; eigenphases $\{\theta_k\}$ of $\bU$.
\end{itemize}

\subsection*{Compare}
\begin{itemize}[leftmargin=2em]
\item Gap surrogates: spectral gap $\lambda_2$ (connectivity); largest eigenvalue $\lambda_{\max}$ (stability range).
\item ST proxy: distance of phase law to Sato--Tate $\W_2(\mu,\mu_{\mathrm{ST}})$; KS/CvM statistics with BCa CIs.
\item Invariance checks: $\KS(\spec(\bU),\spec(P_g \bU P_g^\top))$ and permutation-invariant row-moment tuples.
\end{itemize}

\subsection*{Equalize}
\begin{itemize}[leftmargin=2em]
\item Certified smoothing (heat-flow step): $u^+=u-\alpha \bL u$ with $0<\alpha<2/\lambda_{\max}$.
\item Projection-first safety (ACE): weighted-$\ell_1$ projection of parameter proposals with KKT certificates.
\item Lipschitz envelope: $\Lip_{1\to 1}$ upper bound $\mathrm{SlopeUB}$ from linear operator norms and local softmax Jacobians.
\end{itemize}

\section{Core Lemmas and Certificates}
\begin{lem}[Permutation--equivariant unitarization]\label{lem:perm-eq}
For any permutation $P$ and any of the two paths, $\bU(P\bA P^\top)=P\,\bU(\bA)\,P^\top$. Hence the phase multiset $\{\theta_k\}$ is class-invariant under $G$.
\end{lem}

\begin{prop}[Heat-flow contraction certificate]\label{prop:contraction}
Let $u^\perp=u-\bar u\,\one$ where $\bar u=\tfrac1n \one^\top u$. For $u^+=u-\alpha \bL u$ with $0<\alpha<2/\lambda_{\max}$,
\[
\norm{(u^+)^\perp}_2 \;\le\; (1-\alpha \lambda_2)\, \norm{u^\perp}_2.
\]
On violation, rollback and reduce $\alpha$; log $(E[u],\, r_t,\, 1-\alpha\lambda_2)$ where $r_t=\norm{(u^+)^\perp}/\norm{u^\perp}$.
\end{prop}

\begin{prop}[ACE projection with KKT certificate]\label{prop:ace}
Given weights $\omega_k>0$ and budget $T>0$, the Euclidean projection of $v\in\R^m$ onto $\Omega=\{x: \sum_k \omega_k \abs{x_k}\le T\}$ is
\[
x_k^\star=\operatorname{sign}(v_k)\,\max\{\abs{v_k}-\tau\,\omega_k,\,0\},\qquad \tau\ge 0,
\]
with $\tau$ chosen so that $\sum_k \omega_k\abs{x_k^\star}=T$ if active. Feasibility and complementary slackness hold; a computable dual lower bound certifies the gap.
\end{prop}

\begin{prop}[Softmax Jacobian \& Slope bound]\label{prop:softmax}
For $s=\softmax(z)$, the Jacobian $J=\diag(s)-ss^\top$ obeys $\norm{J}_{1\to 1}\le \max_i 2 s_i(1-s_i)\le \tfrac12$. If layer-wise linear maps satisfy $\norm{W^{(\ell)}}_{1\to 1}\le B_\ell$ and softmax uses have runtime upper bounds $\mathrm{SoftmaxUB}^{(\ell)}\in[0,\tfrac12]$, then an end-to-end envelope is
\[
\mathrm{SlopeUB} \;=\; \Big(\prod_\ell B_\ell\Big)\,\Big(\prod_\ell \mathrm{SoftmaxUB}^{(\ell)}\Big).
\]
\end{prop}

\section{Bridging Automorphy to Graph Energetics}
\begin{defn}[Graph from attention]
Given $\tilde A$ and $\bB$ as above, define $\bW=\tfrac12(\bB+\bB^\top)$, $\bL=\bD-\bW$. We say $G=(V,\bW)$ is the \emph{attention graph}.
\end{defn}

\begin{prop}[Energy invariance under $G$-actions]\label{prop:energy-inv}
If the mask and logits are $G$-equivariant so that $\bW=P_g \bW P_g^\top$ for all $g\in S\subseteq G$, then for any $u$ and $g\in S$,
\(
E[u\circ g] = E[u].
\)
\end{prop}
\begin{proof}
$E[u\circ g]=\tfrac12 (P_g u)^\top \bL (P_g u)=\tfrac12 u^\top (P_g^\top \bL P_g) u= \tfrac12 u^\top \bL u$.
\end{proof}

\begin{prop}[Compatibility of Compare-layer metrics]\label{prop:metrics}
Under Lem.~\ref{lem:perm-eq}, (i) phase-law metrics (e.g., $\W_2$ to Sato--Tate, $\KS$) are class functions; (ii) $\lambda_2(\bL)$ and row-moment summaries of $\tilde A$ are permutation-invariant. Thus AU--Triad comparisons commute with automorphic actions.
\end{prop}

\section{Composite Pass/Fail Gate}
Set thresholds $\tau_{\mathrm{ST}}, \Delta_{\mathrm{CI}}, \varepsilon_{\mathrm{unit}}, \tau_{\mathrm{Slope}}, \tau_{\mathrm{Gap}}$.
A step is \emph{accepted} if all hold:
\begin{align*}
&\textbf{(ST)} && \mathrm{median}\,\W_2(\mu_\theta,\mu_{\mathrm{ST}})\le \tau_{\mathrm{ST}} \ \text{and CI width}\le \Delta_{\mathrm{CI}},\\
&\textbf{(Perm)} && \KS(\spec(\bU),\spec(P_g \bU P_g^\top)) \le \varepsilon_{\mathrm{perm}} \ \text{(for all sampled $g$)},\\
&\textbf{(Unit)} && \norm{\bU^*\bU-\Id}_{1\to 1} \le \varepsilon_{\mathrm{unit}},\\
&\textbf{(ACE)} && \text{ACE projection feasible and dual gap}\le \tau_{\mathrm{Gap}},\\
&\textbf{(Slope)} && \mathrm{SlopeUB}\le \tau_{\mathrm{Slope}},\\
&\textbf{(Heat)} && r_t \le 1-\alpha \lambda_2 \quad\text{(Prop.~\ref{prop:contraction}).}
\end{align*}
On any failure: rollback, shrink the offending knob (e.g., $\alpha\mapsto \alpha/2$, tighten budgets), and retry once.

\section{Integration Losses and Objectives}
\paragraph{CSL invariance loss.}
Let $S\subseteq G$ be a minibatch of actions. For an invariant tuple $F_{\mathrm{inv}}(\cdot)$ (e.g., $\spec(\bU)$ plus sorted row-moments),
\[
\mathcal{L}_{\mathrm{CSL}}(\bA) \;=\; \frac{1}{|S|}\sum_{g\in S} \Delta\big(F_{\mathrm{inv}}(\bA^{(g)}),\,F_{\mathrm{inv}}(\bA)\big),\qquad \bA^{(g)}:=P_g \bA P_g^\top.
\]

\paragraph{Graph energy regularization.}
Optionally penalize high-frequency activity w.r.t.\ the attention graph via
\[
\mathcal{R}_{\mathrm{HF}}(u) \;=\; E[u] \;=\; \tfrac12 u^\top \bL u,
\]
or constrain outputs to a low-frequency cone $F_k=\mathrm{span}\{v_1,\dots,v_k\}$ using the projector $\Pi_{F_k}$.

\paragraph{Unitary spectral proxy.}
Compare $\mu_\theta$ (empirical eigenphase law of $\bU$) to a target $\mu_{\mathrm{ST}}$ using $\W_2$/KS/CvM with preregistered seeds and bootstrap CIs.

\section{Algorithmic Playbook (Projection-First + Certificates)}
\begin{algorithm}[H]
\caption{AU--Triad $\oplus$ Automorphic Learning (one certified step)}
\begin{algorithmic}[1]
\State \textbf{Inputs:} $Q,K$, primes $(p_1,p_2)$, CRT map $\varphi$, mask $M$, budgets $(\omega,\ T)$, thresholds, step $\alpha$.
\State \textbf{Logits:} $L \gets QK^\top/\sqrt d + \alpha M - \beta(\one-M)$; $\tilde A \gets \softmax(L)$.
\State \textbf{Sinkhorn:} $\bB\gets \mathrm{Sinkhorn}(\mathrm{ReLU}(\tilde A)+\varepsilon)$; \textbf{check} positivity residual.
\State \textbf{Unitarize:} $\bU\gets \exp\big(\pi i(\bB-\bJ)\big)$ (or Cayley); \textbf{log} unitary residual.
\State \textbf{Compare:} evaluate $\W_2(\mu_\theta,\mu_{\mathrm{ST}})$, KS/CvM, permutation checks $g\sim S$.
\State \textbf{Graphize:} $\bW\gets (\bB+\bB^\top)/2$, $\bL\gets \bD-\bW$; estimate $(\lambda_2,\lambda_{\max})$.
\State \textbf{Equalize:} $u^+\gets u-\alpha \bL u$; compute $r_t$ and bound $1-\alpha\lambda_2$; \textbf{require} $r_t$ within bound.
\State \textbf{Projection (ACE):} project parameter proposal $v$ onto $\sum_k \omega_k|x_k|\le T$; \textbf{log} $(\tau,\text{gap})$.
\State \textbf{Slope envelope:} compute $\mathrm{SoftmaxUB}$ and $\mathrm{SlopeUB}$; \textbf{require} $\mathrm{SlopeUB}$ within ceiling.
\If{all gates pass} \State \textbf{accept} step and append audit record; \Else \State \textbf{rollback}, shrink step/budgets, retry once. \EndIf
\end{algorithmic}
\end{algorithm}

\paragraph{Notation recap.}
$\tilde A$ row-stochastic attention; $\bB$ $\varepsilon$-stabilized bistochastic matrix; $\bU$ unitarization; $\bW$ symmetrized weights; $\bL$ Laplacian; $E[u]$ Dirichlet energy; $\lambda_2$ spectral gap; $\mathrm{SoftmaxUB}\in[0,\tfrac12]$; $\mathrm{SlopeUB}$ envelope; $P_g$ permutation for $g\in G$.

\section{Proof Sketches (selected)}
\paragraph{Lemma~\ref{lem:perm-eq}.}
Sinkhorn scaling is permutation-equivariant, i.e.\ $\mathrm{Sinkhorn}(P X P^\top)=P\,\mathrm{Sinkhorn}(X)\,P^\top$ and $\bJ$ commutes with permutations. Matrix exponential and Cayley transforms commute with conjugation; thus $\bU(P\bA P^\top)=P\bU(\bA)P^\top$.

\paragraph{Prop.~\ref{prop:contraction}.}
Diagonalize $\bL=V\Lambda V^\top$; the update scales the $k$-th eigen-coordinate by $(1-\alpha\lambda_k)$. The constant mode is fixed; all others contract by at most $1-\alpha\lambda_2$ when $0<\alpha<2/\lambda_{\max}$.

\paragraph{Prop.~\ref{prop:softmax}.}
$J=\diag(s)-ss^\top$ is a doubly-centered covariance; $\ell_1\to \ell_1$ norm is bounded by $\max_i 2s_i(1-s_i)$, maximized at $1/2$.

\section{Extensions}
\paragraph{Sheaf/presheaf interface.}
Presheaf over primes (one object per $p$ and layer) induces parameter-tying maps and token restrictions; AU--Triad's sheaf Laplacians give consistency penalties $\sum\limits_{p,q,W}\big\lVert E_p(\theta)\rvert_W - E_q(\theta)\rvert_W\big\rVert$.

\paragraph{Transport and DEC.}
Optionally replace heat-flow equalization by entropic JKO steps (optimal transport) or Hodge/connection Laplacians for bundle-valued signals; the same contraction and projection certificates apply with operator substitutions.

\section{Compliance Checklist}
\begin{enumerate}[leftmargin=2em,label=(\alph*)]
\item Validate symmetry/shape; log seeds and masks.
\item Estimate or bound $(\lambda_2,\lambda_{\max})$; choose $0<\alpha<2/\lambda_{\max}$.
\item Evaluate contraction certificate $r_t\le 1-\alpha\lambda_2$; rollback on fail.
\item Run ST, permutation, unitary residual, ACE, and Slope gates; all must pass.
\item Maintain machine-readable audit trail of all quantities above.
\end{enumerate}
