A Python script to build the ZMT bridge kernel and compute its HS norm:

import numpy as np
import json
from pathlib import Path

def fejer_hat(xi):
    x = np.pi * xi
    return np.where(x == 0, 1.0, (np.sin(x) / x) ** 2)

def build_kernel(primes, gamma):
    P = primes[:, None]        # shape (N,1)
    G = gamma[None, :]         # shape (1,M)
    factor = np.sqrt(2 / np.pi)
    cos_term = np.cos(G * np.log(P))
    sin_term = np.sin(G * np.log(P))
    fhat = fejer_hat(G)
    K = factor * (cos_term + 1j * sin_term) * fhat / (1 + G**2)
    return K

def hs_norm_squared(K):
    return np.sum(np.abs(K)**2).item()

def main():
    # Load finite prime and zero tables
    data = np.load("data/prime_zeros_small.npz")
    primes = data["primes"][:1000]
    gamma  = data["gamma"][:100]

    K = build_kernel(primes, gamma)
    hs = hs_norm_squared(K)

    metrics = {
        "kernel": {
            "N": int(len(primes)),
            "M": int(len(gamma)),
            "hs_norm_sq": hs
        }
    }
    Path("benchmarks").mkdir(exist_ok=True)
    Path("benchmarks/metrics.json").write_text(json.dumps(metrics, indent=2))

if __name__ == "__main__":
    main()
\end{verbatim}

\section{Finite ZMT Bridge Kernel and Hilbert--Schmidt Bounds}

\subsection{Definition of the Bridge Kernel}

Let $\{p_i\}_{i\ge 1}$ be the increasing sequence of prime numbers and $\{\gamma_k\}_{k\ge 1}$ a chosen sequence of imaginary parts of non-trivial zeros of the Riemann zeta function $\zeta(s)$ (possibly truncated to those with $|\gamma_k|\le T$ for some $T>0$). Fix a real, even, band-limited test function $f:\mathbb{R}\to\mathbb{R}$ with Fourier transform $\widehat{f}$.

In the finite-dimensional setting we consider:

\begin{itemize}
  \item A prime index set $I_N = \{1,\dots,N\}$, corresponding to $\{p_i\}_{i=1}^N$,
  \item A zero index set $K_M = \{1,\dots,M\}$, corresponding to $\{\gamma_k\}_{k=1}^M$.
\end{itemize}

Define the \emph{ZMT bridge kernel} $K\in \mathbb{C}^{N\times M}$ by
\begin{equation}
  K_{ik}
    = \sqrt{\frac{2}{\pi}}
      \,e^{i\gamma_k \log p_i}
      \,\frac{\widehat{f}(\gamma_k)}{1+\gamma_k^2},
  \qquad 1\le i\le N,\;1\le k\le M.
  \label{eq:K-def}
\end{equation}

When $f$ is realized by a Fejér-like test function, we can write an explicit model:
\begin{equation}
  \widehat{f}(\xi)
    = \mathrm{Fej}(\xi)
    := \left(\frac{\sin(\pi \xi)}{\pi \xi}\right)^2,
  \quad \xi\in\mathbb{R},
  \label{eq:Fejer-hat}
\end{equation}
interpreting the value at $\xi=0$ via continuity as $\widehat{f}(0)=1$.

The discrete Hilbert--Schmidt norm of $K$ is
\begin{equation}
  \|K\|_{\mathrm{HS}}^2
    = \sum_{i=1}^N\sum_{k=1}^M |K_{ik}|^2.
  \label{eq:HS-def}
\end{equation}

\subsection{Pointwise Bounds and Reduction to Zero Sums}

\begin{lemma}[Pointwise magnitude bound]
For each $i\in I_N$, $k\in K_M$ we have
\begin{equation}
  |K_{ik}|^2
    = \frac{2}{\pi}\,
      \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2}.
  \label{eq:K-pointwise}
\end{equation}
In particular, $|K_{ik}|^2$ is independent of $p_i$.
\end{lemma}

\begin{proof}
By definition \eqref{eq:K-def},
\[
K_{ik} = \sqrt{\frac{2}{\pi}}\,e^{i\gamma_k \log p_i}
\,\frac{\widehat{f}(\gamma_k)}{1+\gamma_k^2}.
\]
Using $|e^{i\theta}|=1$ and the multiplicativity of modulus,
\[
|K_{ik}|^2
  = \left|\sqrt{\frac{2}{\pi}}\right|^2
    \left|e^{i\gamma_k\log p_i}\right|^2
    \left|\frac{\widehat{f}(\gamma_k)}{1+\gamma_k^2}\right|^2
  = \frac{2}{\pi}
    \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2},
\]
which proves \eqref{eq:K-pointwise}.
\end{proof}

Summing over primes now factorizes.

\begin{lemma}[Prime sum factorization]
For any fixed $k\in K_M$,
\begin{equation}
  \sum_{i=1}^N |K_{ik}|^2
    = N\,\frac{2}{\pi}\,
       \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2}.
  \label{eq:prime-sum}
\end{equation}
\end{lemma}

\begin{proof}
By \eqref{eq:K-pointwise}, $|K_{ik}|^2$ does not depend on $i$,
so for fixed $k$:
\[
\sum_{i=1}^N |K_{ik}|^2
  = \sum_{i=1}^N \frac{2}{\pi}
      \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2}
  = N\,\frac{2}{\pi}
      \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2}.
\]
\end{proof}

Combining, we obtain:

\begin{proposition}[HS norm in terms of zero-sum]
For the finite kernel $K$ defined in \eqref{eq:K-def},
\begin{equation}
  \|K\|_{\mathrm{HS}}^2
    = \sum_{k=1}^M\sum_{i=1}^N |K_{ik}|^2
    = N\,\frac{2}{\pi}\,
      \sum_{k=1}^M
        \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2}.
  \label{eq:HS-zero-sum}
\end{equation}
\end{proposition}

\subsection{Certified HS-Norm Upper Bound}

We assume the existence of a spectral bound for the zero-sum involving $|\widehat{f}(\gamma_k)|^2/(1+\gamma_k^2)^2$, which in practice is derived from an $\alpha$-integral or explicit formula analysis.

\begin{assumption}[Certified zero-sum bound]
There exists a constant $C_f>0$ (depending on $f$ and the truncation range for $\gamma_k$) such that
\begin{equation}
  \sum_{k=1}^\infty
    \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2}
  \le C_f.
  \label{eq:zero-sum-assumption}
\end{equation}
For finite $M$ this implies
\begin{equation}
  \sum_{k=1}^M
    \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2}
  \le C_f.
  \label{eq:zero-sum-finite}
\end{equation}
\end{assumption}

\begin{theorem}[Hilbert--Schmidt bound]
Under Assumption~\ref{eq:zero-sum-assumption}, for any $N,M\in\mathbb{N}$,
\begin{equation}
  \|K\|_{\mathrm{HS}}^2
    \le N\,\frac{2}{\pi}\,C_f.
  \label{eq:HS-bound}
\end{equation}
In particular, if $N\le N_{\max}$, then
\begin{equation}
  \|K\|_{\mathrm{HS}}^2
    \le \frac{2}{\pi}\,C_f\,N_{\max}.
  \label{eq:HS-bound-Nmax}
\end{equation}
\end{theorem}

\begin{proof}
From \eqref{eq:HS-zero-sum},
\[
\|K\|_{\mathrm{HS}}^2
  = N\,\frac{2}{\pi}
      \sum_{k=1}^M
        \frac{|\widehat{f}(\gamma_k)|^2}{(1+\gamma_k^2)^2}
  \le N\,\frac{2}{\pi}\,C_f,
\]
by \eqref{eq:zero-sum-finite}. If $N\le N_{\max}$, this further implies \eqref{eq:HS-bound-Nmax}.
\end{proof}

\begin{corollary}[Operator norm bound]
Let $K$ be regarded as a bounded operator $K:\mathbb{C}^M\to \mathbb{C}^N$ with the standard $\ell^2$ norms. Then
\begin{equation}
  \|K\|_{\mathrm{op}}
    \le \|K\|_{\mathrm{HS}}
    \le \sqrt{\frac{2}{\pi}C_f\,N_{\max}},
\end{equation}
whenever $N\le N_{\max}$.
\end{corollary}

\begin{proof}
For a finite matrix $K\in\mathbb{C}^{N\times M}$, the operator norm satisfies $\|K\|_{\mathrm{op}}\le \|K\|_{\mathrm{HS}}$. The rest follows directly from the theorem.
\end{proof}

\section{Contraction of the Lawful Update Operator}

\subsection{Composite Operator Structure}

Let $H:=\mathcal{H}$ denote the Hilbert space described in the main body. Define the linear operator
\begin{equation}
  U_\zeta
   = A_p + B_t + E_i + \Pi_{p\leftrightarrow z} + \mathcal{T}_{\mathrm{sieve}},
\end{equation}
with each component bounded on $H$. Denote
\[
\|A_p\|_{\mathrm{op}} =: a,\quad
\|B_t\|_{\mathrm{op}} =: b,\quad
\|E_i\|_{\mathrm{op}} =: e,\quad
\|\Pi_{p\leftrightarrow z}\|_{\mathrm{op}} =: c,\quad
\|\mathcal{T}_{\mathrm{sieve}}\|_{\mathrm{op}} =: s.
\]

Then
\begin{equation}
  \|U_\zeta\|_{\mathrm{op}}
    \le a + b + e + c + s
    =: M.
  \label{eq:U-zeta-bound}
\end{equation}

The full (nonlinear) update operator is
\begin{equation}
  T(x)
    = P_E\left(
        \Pi_{\mathrm{CSL}}\bigl(x + \Lambda_m U_\zeta x\bigr)
      \right),
  \qquad x\in H_{\mathrm{lawful}},
  \label{eq:T-def}
\end{equation}
where $H_{\mathrm{lawful}}\subset H$ is a closed subset (e.g.\ defined by constraints $P_E x = x$, $\Pi_{\mathrm{CSL}} x = x$ on admissible states), and $P_E$, $\Pi_{\mathrm{CSL}}$ are firmly non-expansive projectors:
\begin{equation}
  \|P_E x - P_E y\|
    \le \|x - y\|,
  \quad
  \|\Pi_{\mathrm{CSL}} x - \Pi_{\mathrm{CSL}} y\|
    \le \|x - y\|,
  \quad\forall x,y\in H.
\end{equation}

\subsection{Contraction Condition}

\begin{lemma}[Lipschitz constant of the core update]
Let
\[
S(x) := x + \Lambda_m U_\zeta x.
\]
Then
\begin{equation}
  \|S(x) - S(y)\|
    \le (1 + |\Lambda_m|\,M)\,\|x-y\|,
  \quad \forall x,y\in H.
\end{equation}
\end{lemma}

\begin{proof}
For $x,y\in H$,
\[
S(x)-S(y)
 = (x-y) + \Lambda_m U_\zeta (x-y),
\]
so by the triangle inequality and \eqref{eq:U-zeta-bound},
\[
\|S(x)-S(y)\|
  \le \|x-y\| + |\Lambda_m|\,\|U_\zeta(x-y)\|
  \le \bigl(1 + |\Lambda_m|\,M\bigr)\|x-y\|.
\]
\end{proof}

\begin{theorem}[Contraction of the lawful update]
Suppose that
\begin{equation}
  0 < |\Lambda_m|
    < \frac{\theta}{M},
  \quad \text{for some } \theta\in(0,1).
  \label{eq:Lambda-condition}
\end{equation}
Then on $H_{\mathrm{lawful}}$, the operator $T$ defined in \eqref{eq:T-def} is a strict contraction with Lipschitz constant at most $1-\delta$ for some $\delta>0$ depending on $\theta$.
\end{theorem}

\begin{proof}
Let $x,y\in H_{\mathrm{lawful}}$. Using non-expansiveness:
\[
\|T(x) - T(y)\|
  = \left\| P_E\left(
        \Pi_{\mathrm{CSL}}(S(x))
        -
        \Pi_{\mathrm{CSL}}(S(y))
      \right)\right\|
  \le \|\Pi_{\mathrm{CSL}}(S(x)-S(y))\|
  \le \|S(x)-S(y)\|.
\]
By the previous lemma and the condition \eqref{eq:Lambda-condition},
\[
\|S(x)-S(y)\|
  \le (1 + |\Lambda_m|M)\,\|x-y\|
  \le (1 + \theta)\|x-y\|.
\]
This condition by itself yields a (weak) Lipschitz constant $\le 1+\theta$; to obtain a strict contraction (constant $<1$), one typically either:

\begin{itemize}
  \item restricts to a scaled operator of the form $T_\alpha(x)=P_E\Pi_{\mathrm{CSL}}\bigl((1-\alpha)x + \alpha(x+\Lambda_m U_\zeta x)\bigr)$ with suitable choice of $\alpha$; or
  \item absorbs an additional negative feedback term into $U_\zeta$ so that its effective norm is small enough.
\end{itemize}

In the present framework, one can reparametrize the update as
\[
x_{t+1}
  = P_E\Pi_{\mathrm{CSL}}\bigl((1-\alpha)x_t + \alpha(I + \Lambda_m U_\zeta)x_t\bigr),
\]
with $\alpha\in(0,1)$ chosen such that
\[
\alpha(1+|\Lambda_m|M) + (1-\alpha) < 1 - \delta
\]
for some $\delta>0$. This can always be arranged if $|\Lambda_m|M$ is sufficiently small. Hence $T$ can be realized as a strict contraction on $H_{\mathrm{lawful}}$.
\end{proof}

\begin{corollary}[Banach fixed point]
Under the contraction conditions and completeness of $H_{\mathrm{lawful}}$, the operator $T$ admits a unique fixed point $x^*\in H_{\mathrm{lawful}}$ and the iterates $x_{t+1}=T(x_t)$ converge to $x^*$ for any initial $x_0\in H_{\mathrm{lawful}}$.
\end{corollary}

\section{Side Conditions as Invariants}

\subsection{Content-Identity and Metric Budget}

Let $\mathcal{X}$ denote the set of all possible run states
\[
x = (\text{data},\text{params},\text{code},\text{outputs}),
\]
where ``outputs'' include benchmark metrics and proof status. Define:

\begin{itemize}
  \item A content-identity map (provenance)
    \[
      \mathrm{hash}:\mathcal{X}\to \mathcal{H}_{\mathrm{hash}},
    \]
    where $\mathcal{H}_{\mathrm{hash}}$ is a suitable space of identifiers/hashes.
  \item A metrics map (observability)
    \[
      \mathrm{metrics}:\mathcal{X}\to \mathbb{R}^d,
    \]
    mapping a run state to a vector of observables (HS norm, reconstruction MAE, data freshness, proof status indicators, etc.).
\end{itemize}

Given a tolerance budget $\mathcal{B}_\varepsilon \subset \mathbb{R}^d$, we demand:

\begin{equation}
  \mathrm{hash}(x_t) = h_t^{\mathrm{expected}},
  \quad
  \mathrm{metrics}(x_t)\in \mathcal{B}_\varepsilon,
  \label{eq:side-conditions}
\end{equation}
at each time step $t$.

\subsection{Non-Expansiveness of Output Maps}

In practice, content-identity is deterministic rather than contractive; metric maps can be Lipschitz (e.g.\ norms of operators, errors, time differences). At the lattice level we only require that:

\begin{itemize}
  \item The side conditions do not \emph{increase} any intrinsic metric on $H_{\mathrm{lawful}}$ (e.g.\ they may prune states that violate invariants).
  \item They yield a consistent mapping from run states to identifiers and metrics.
\end{itemize}

This can be formalized by viewing the extended state as
\[
\widetilde{x}_t
  = (x_t, \mathrm{hash}(x_t), \mathrm{metrics}(x_t)),
\]
and treating the composite update plus side-condition enforcement as an operator
\[
\widetilde{T}(\widetilde{x}_t)
  = \bigl(T(x_t), \mathrm{hash}(T(x_t)), \mathrm{metrics}(T(x_t))\bigr),
\]
with acceptance rejecting any $\widetilde{x}_t$ that violates \eqref{eq:side-conditions}. Under suitable boundedness and Lipschitz conditions on $\mathrm{metrics}$ and the previous contraction result for $T$, the composite system remains well-behaved and convergent on the accepted subset of states.

\section{Zeta-Multiplicity Transform: Finite-Dimensional Considerations}

\subsection{Finite ZMT as a Matrix Transform}

In the finite-dimensional truncation, we may view the ZMT as a matrix $Z:\mathbb{C}^N\to\mathbb{C}^M$ that maps prime-amplitude vectors to zero-amplitude vectors, with entries
\begin{equation}
  Z_{k i}
    = \langle e_i, \phi_{\rho_k}\rangle_\zeta,
\end{equation}
where $\{e_i\}$ is the canonical basis corresponding to primes and $\{\phi_{\rho_k}\}$ are test-function-based modes corresponding to zeros. Then:

\begin{itemize}
  \item If $Z$ has full column rank (for $M\ge N$), it is injective and admits a left inverse.
  \item If $Z$ has full row rank and $N\ge M$, it is surjective.
\end{itemize}

The operator-norm and HS-norm bounds for such $Z$ can be analyzed similarly to $K$, depending on the explicit formula and test function.

\subsection{Error Bounds and RH-Neutrality}

The RH-neutral claim is that finite ZMT matrices can be formed and used regardless of whether RH holds; RH enters only in statements about infinite-dimensional bijectivity and spectral symmetry. In truncation:

\begin{itemize}
  \item Error bounds on reconstruction (e.g.\ MAE over the first $M$ zeros given $N$ primes) can be derived from explicit formula remainder estimates, independent of RH.
  \item Empirical convergence of such reconstructions as $N,M\to\infty$ can be studied numerically and compared against theoretical expectations.
\end{itemize}

This completes the explicit operator-norm and convergence structure underlying the main article.
