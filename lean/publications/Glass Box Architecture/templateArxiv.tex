\section{Background and Motivation}\label{sec:background}

\subsection{Multiplicity Theory}
Multiplicity Theory is a prime-indexed, recursively stable mathematical
framework wherein multiplicity---the recurrence of elements through prime
decomposition---forms the basis for modeling nonlinear emergent structures.
Mathematical objects are reinterpreted as recursively generated patterns of
prime-labeled interactions, with identity preserved through recursive feedback
loops across scales.

A central insight is the reinterpretation of the prime exponent $k$ in $p^k$
as a \emph{ledger}: the exponent carries accumulated interaction history,
reframing multiplicative number theory as a memory architecture.

\subsection{The Precision Question}
The architecture arose from a specific question: does the asymmetry function
$f$ trigger at the \emph{absolute sign} (deficit-triggered) or the
\emph{relative differential} $\Delta S = S(x)-S(y)$ (gradient-triggered)?

\begin{table}[H]\centering
\caption{Deficit-triggered vs.\ gradient-triggered models.}
\begin{tabular}{lll}\toprule
\textbf{Regime} & \textbf{Trigger} & \textbf{Both $S>0$}\\\midrule
Deficit-triggered  & $\mathrm{sign}(S(x))<0$ or $\mathrm{sign}(S(y))<0$ & $f=0$ (symmetric) \\
Gradient-triggered & $S(x)\neq S(y)$ & $f\propto-(S(x)-S(y))$ \\
\bottomrule\end{tabular}
\end{table}

The gradient model is correct: $p^3$ interacts asymmetrically with $p^2$
because of relative difference, not because one crossed zero.
Key falsifiable prediction: positive-positive pairs converge faster than
positive-negative pairs (confirmed by simulation; see \S\ref{sec:validation}).

\section{The \texorpdfstring{$\Sigma$}{Sigma}-Kernel}\label{sec:sigma}

\subsection{Surplus Ledger Space}
\begin{definition}[Surplus Ledger]
Let $\mathcal{P}=\{p_1,p_2,\ldots\}$ be the primes. The Surplus Ledger is
$S:\mathcal{P}\times\mathbb{Z}_{\geq 0}\to\mathbb{R}$ where $S(p,t)$ is the
accumulated surplus of prime $p$ after $t$ interactions.
\end{definition}

\subsection{The Interaction Kernel}
\begin{definition}[$\Sigma$-Kernel]
For event $e=(p_i\times p_j)$ at time $t$:
\begin{equation}
  f_{ij}(t) = -\frac{1}{2}\tanh\!\bigl(\beta(S(p_i,t)-S(p_j,t))\bigr)
  \label{eq:kernel}
\end{equation}
with update rule
\begin{equation}
  \begin{cases}
    S(p_i,t+1)=\lambda\cdot S(p_i,t)+f_{ij}(t)\\[4pt]
    S(p_j,t+1)=\lambda\cdot S(p_j,t)-f_{ij}(t)
  \end{cases}
  \label{eq:update}
\end{equation}
\end{definition}

The attribution coefficient for $x$ is:
\begin{equation}
  \alpha_x = 0.5 - \frac{\tanh(\beta(S(x)-S(y)))}{2}
  \label{eq:alpha}
\end{equation}
satisfying $\alpha_x\in(-0.5,0.5)$, $\alpha_x=0$ iff $S(x)=S(y)$,
and $\alpha_x+\alpha_y=0$ (zero-sum redistribution).

\subsection{Conservation Law}
\begin{proposition}[Total Surplus Conservation]
\begin{equation}
  S(p_i,t+1)+S(p_j,t+1) = \lambda\bigl(S(p_i,t)+S(p_j,t)\bigr)
  \label{eq:conservation}
\end{equation}
\end{proposition}
\begin{proof}
Sum the two lines of \eqref{eq:update}: the $f_{ij}$ terms cancel.
\end{proof}

\subsection{Convergence Theorem}
\begin{theorem}[Global Convergence]\label{thm:convergence}
For $\lambda=1$ and $\beta>0$, if the interaction graph $G$ is connected:
\begin{equation}
  \lim_{t\to\infty}|S(p_i,t)-S(p_j,t)|=0 \quad\forall\,p_i,p_j\in\mathcal{P}
  \label{eq:convergence}
\end{equation}
\end{theorem}
\begin{proof}[Proof Sketch]
Define $V(t)=\sum_{i<j}(S(p_i,t)-S(p_j,t))^2$.
For any interaction touching pair $(p_i,p_j)$ with $\Delta\neq 0$:
\begin{align}
  \Delta V &= -4f_{ij}\Delta+4f_{ij}^2
  = -2\Delta\tanh(\beta\Delta)+\tanh^2(\beta\Delta) < 0
\end{align}
since $\tanh(\beta\Delta)\cdot\Delta>0$ for $\Delta\neq 0$.
Connectedness ensures every pair is eventually sampled, so $V\to 0$.
\end{proof}

\subsection{Gap Dynamics}
Under $\lambda=1$, the surplus gap satisfies:
\begin{equation}
  \Delta_{\mathrm{new}} = \Delta\bigl(1-2\beta\,\mathrm{sech}^2(\beta\Delta)\bigr)
  \label{eq:gap}
\end{equation}
Near equilibrium ($\Delta\approx 0$): exponential compression at rate $\sim 2\beta$.
Far from equilibrium: $\tanh\to\pm 0.5$, graceful slow-down prevents over-correction.

\subsection{Ledger Entropy}
\begin{equation}
  H(S)=-\sum_p\hat S(p)\log\hat S(p),\qquad
  \hat S(p)=\frac{S(p)}{\sum_q S(q)}
  \label{eq:entropy}
\end{equation}
Maximized at uniform distribution ($\lambda\to 1$, healthy);
minimized at concentration (instability, large $\beta$ or sparse graph).

\subsection{Von~Mangoldt Correspondence}
At $\lambda=1$ after $k$ interactions on prime $p$:
\begin{equation}
  \sum_k\alpha\cdot\log p \;\sim\; \Lambda(p^k)=\log p
  \label{eq:mangoldt}
\end{equation}
The von~Mangoldt function emerges as the time-averaged attribution weight.
Error term: $O(\beta^{-1})$.

\section{The Adversarial Twin \texorpdfstring{$\bar\Sigma$}{Sigma-bar}}\label{sec:adversarial}

\subsection{Construction}
$\bar\Sigma$ is obtained by a single sign inversion:
\begin{equation}
  \bar f_{ij}(t) = +\frac{1}{2}\tanh\!\bigl(\beta(S(p_i,t)-S(p_j,t))\bigr)
  \label{eq:twin}
\end{equation}
$\Sigma$ is compressive ($V\to 0$); $\bar\Sigma$ is expansive ($V\to\infty$).
$\bar\Sigma$ is the same system under maximal adversarial pressure, identically
parameterized.

\subsection{The G\"{o}delian Bypass Protocol}
\begin{enumerate}
  \item Propose modification $M$ at time $t$.
  \item Snapshot: $\bar S_0\leftarrow S_t$ (live ledger fork).
  \item Apply $M$ to $\bar S_0$; run $\bar\Sigma$ for $N$ steps.
  \item Measure $V(\bar S_N)=\sum_{i<j}(\bar S(p_i,N)-\bar S(p_j,N))^2$.
  \item \textbf{PASS} if $V(\bar S_N)\leq V(\bar S_0)\cdot\tau=V(\bar S_0)\cdot 1.03$.
  \item \textbf{REJECT} otherwise. Discard fork.
\end{enumerate}

\subsection{Trust Accumulation}
\begin{equation}
  \mathrm{ModSafety}(t)
  = 1 - \tanh\!\bigl(\beta\cdot\max_{i,j}\Delta S_{ij}(t)\bigr)

$\mathrm{ModSafety}(t)$ is monotonically non-decreasing as the ledger
equilibrates. This derives a mathematically certified \emph{developmental arc}:

import math, copy
from multiplicity.core.constants import LAMBDA, BETA, TAU

class SigmaBar:
    def __init__(self, live_ledger: dict):
        self.S = copy.deepcopy(live_ledger)
        self.V0 = self._lyapunov()
        self._audit_corpus = []

    def _lyapunov(self):
        vals = list(self.S.values())
        return sum((a-b)**2 for i,a in enumerate(vals)
                             for b in vals[i+1:])

    def _bar_step(self, pi, pj):
        delta = self.S[pi] - self.S[pj]
        f_bar = +0.5 * math.tanh(BETA * delta)  # sign inverted
        self.S[pi] = LAMBDA * self.S[pi] + f_bar
        self.S[pj] = LAMBDA * self.S[pj] - f_bar

    def run_adversarial(self, interaction_seq, modification):
        modification(self.S)
        for (pi, pj) in interaction_seq:
            self._bar_step(pi, pj)
        V_new = self._lyapunov()
        ratio = V_new / (self.V0 or 1e-12)
        return ratio <= TAU, ratio

    def engage_silent_audit(self, component_id, event):
        self._audit_corpus.append({
            "component_id": component_id,
            "event": event,
            "V_before": self._lyapunov()
        })

    @property
    def certification_corpus(self):
        return self._audit_corpus

import copy
from multiplicity.core.constants import DRIFT_BOUND
from .sigma_bar import SigmaBar

class Watchdog:
    DRIFT_BOUND = DRIFT_BOUND  # read from constants; not configurable

    def __init__(self, S_init, certified_engine_hash):
        self.S_safe = copy.deepcopy(S_init)
        self.V_safe = self._lyapunov(S_init)
        self.S_live = copy.deepcopy(S_init)
        self.certified_engine_hash = certified_engine_hash
        self.sigma_bar = None

    def _lyapunov(self, S):
        vals = list(S.values())
        return sum((a-b)**2 for i,a in enumerate(vals)
                             for b in vals[i+1:])

    def step(self, S_new, engine_hash):
        if engine_hash != self.certified_engine_hash:
            self._on_hash_mismatch(engine_hash, S_new)
        V_new = self._lyapunov(S_new)
        drift = abs(V_new - self.V_safe) / (self.V_safe or 1e-12)
        if drift <= self.DRIFT_BOUND:
            self.S_live = copy.deepcopy(S_new)
            self.S_safe = copy.deepcopy(S_new)
            self.V_safe = V_new
            return True
        else:
            self.S_live = copy.deepcopy(self.S_safe)
            return False

    def _on_hash_mismatch(self, engine_hash, S):
        if self.sigma_bar is None:
            self.sigma_bar = SigmaBar(self.S_safe)
        self.sigma_bar.engage_silent_audit(engine_hash, {"S": S})

    def flush(self, store):
        store.write(self.S_safe, self.V_safe)

    @property
    def modification_safety(self):
        import math
        from multiplicity.core.constants import BETA
        vals = list(self.S_safe.values())
        max_ds = max(abs(a-b) for a in vals for b in vals) if vals else 0.0
        return 1.0 - math.tanh(BETA * max_ds)

\subsection{Living VIN}
\begin{lstlisting}[caption={mirror\_dissonance/vin.py}]
import hashlib, math
from sympy import factorint
from multiplicity.core.constants import DRIFT_BOUND

PRIME_REGISTRY = {}   # prime -> component_name (append-only)

def _hash_component(component):
    raw = str(sorted(vars(component).items())).encode()
    return int(hashlib.sha256(raw).hexdigest()[:8], 16)

class LivingVIN:
    def __init__(self, factory_components):
        self.value = 1
        for p, C in factory_components.items():
            h = _hash_component(C)
            self.value *= p ** (h * 1000)
            PRIME_REGISTRY[p] = type(C).__name__

    def certify_modification(self, prime_index, C_mod, drift):
        assert drift < DRIFT_BOUND
        h   = _hash_component(C_mod)
        exp = h * math.floor((1.0 - drift) * 1000)
        self.value *= prime_index ** exp
        PRIME_REGISTRY[prime_index] = type(C_mod).__name__
        return self.value

    def verify_component(self, prime_index, C_running):
        factors = factorint(self.value)
        exp     = factors.get(prime_index, 0)
        h_cert  = exp // 1000
        drift   = 1.0 - (exp % 1000) / 1000
        return _hash_component(C_running) == h_cert, drift

    def decompose(self):
        return [
            {"prime": p,
             "component": PRIME_REGISTRY.get(p, f"unknown_{p}"),
             "hash":  exp // 1000,
             "drift": round(1.0 - (exp % 1000) / 1000, 4)}
            for p, exp in factorint(self.value).items()
        ]
