\section{Architecture at a Glance}

\subsection{Purpose}
Close the loop between clinical care and learning. One data plane. Two experience skins. Shared governance.

\subsection{System Tenets}
\begin{enumerate}
  \item \textbf{Single data plane:} Intrinsica is the source of truth for signals, features, risk vectors, and provenance.
  \item \textbf{Dual experience skins:} Every learning object renders in \emph{patient education} or \emph{student learning} mode with identical IDs and policies.
  \item \textbf{Privacy-by-design:} De-identification on egress to education; re-identification only inside the S2H enclave.
  \item \textbf{Deterministic traceability:} Every write is purpose-bound, signed, and ledger-hashed with explicit policy versioning.
  \item \textbf{Offline-first option:} Sovereign sites run local bundles and delayed sync with conflict resolution.
\end{enumerate}

\subsection{Layer Stack}
\paragraph{Experience Layer}
XR ``MaaS'' modules, ULMS, and S2H tele-care. Dual skins: patient education \(|\) student learning.
\begin{itemize}
  \item \textbf{Primary responsibilities:} Deliver interventions and lessons; collect attempts, time-on-task, CSL state; surface adherence and outcomes.
  \item \textbf{Inputs:} Risk vectors, personalization hints, content policies, cohort assignments.
  \item \textbf{Outputs:} Telemetry (\texttt{attempts}, \texttt{latency\_ms}, CSL load/gate, TCI), completions, dropouts, consented survey responses.
  \item \textbf{Interfaces:} ULMS Webhooks (\texttt{/hooks/attempted}, \texttt{/hooks/module-complete}, \texttt{/hooks/csl-change}); FHIR Subscriptions for survey and vital-signs.
  \item \textbf{SLOs:} p50 render \(<150\) ms (cloud), p50 input-to-ack \(<300\) ms; sovereign sites \(<24\) h sync tolerance.
  \item \textbf{Failure modes:} Network loss, policy cache miss, stale content; \emph{backoffs:} CSL safe-mode, local queue with replay, version-pinned bundles.
\end{itemize}

\paragraph{Services Layer}
Personalization, content gating (CSL), Reciprocity balancer, \(\pi\)Līla progression, analytics.
\begin{itemize}
  \item \textbf{Personalization:} Maps risk vectors and learner state to module parameters, item ordering, intensity bounds.
  \item \textbf{CSL Gating:} Computes load score and gate state; enforces cooldowns; emits \texttt{csl.state.change}.
  \item \textbf{Reciprocity:} Maintains cohort energy balance; rate-limits group asks; emits \texttt{reciprocity.delta}.
  \item \(\boldsymbol{\pi}\)\textbf{Līla:} Tracks milestones along growth paths; emits \texttt{pilila.milestone}.
  \item \textbf{Analytics:} Risk deltas, adherence, engagement, equity; exports de-ID aggregates to dashboards.
  \item \textbf{SLOs:} p50 personalization \(<100\) ms per request; CSL decisions idempotent within a window; analytics windows hourly (cloud) or daily (sovereign).
\end{itemize}

\paragraph{Data/AI Layer}
Intrinsica lakehouse, feature store, hybrid ML, FHIR/HL7 IO, de-ID stream, ledger.
\begin{itemize}
  \item \textbf{Ingest:} Wearables, EHR, survey data; schema-on-write for FHIR; schema-on-read for analytics.
  \item \textbf{Feature store:} Versioned features with provenance; training/serving skew monitors.
  \item \textbf{Modeling:} Batch and streaming models for sleep, stress, adherence risk, engagement propensity.
  \item \textbf{De-ID stream:} Tokenization, generalization, purpose tagging, policy attach, hashing, emission to education topics.
  \item \textbf{Ledger:} Permissioned, append-only; stores hashes and policy versions; identities via DIDs.
  \item \textbf{SLOs:} p95 ingest-to-feature materialization \(<5\) min; online inference \(<80\) ms p50; ledger commit \(<1\) s.
\end{itemize}

\paragraph{Interop Layer}
FHIR R4 and HL7 v2 where required; LTI 1.3 + xAPI for ULMS; webhooks for events; W3C DIDs for ledger IDs.
\begin{itemize}
  \item \textbf{Clinical objects:} \texttt{Patient, Group, Device, Observation, Procedure, MedicationStatement, QuestionnaireResponse, Consent, Provenance, ResearchStudy, ResearchSubject}.
  \item \textbf{Education traces:} xAPI statements and ULMS webhooks mapped to de-ID event schemas.
  \item \textbf{Subscriptions:} Criteria-based FHIR Subscriptions for survey and vital-signs; rest-hook delivery with auth-bound headers.
  \item \textbf{Identity:} DID-bound JWTs; mTLS between zones; key rotation with audit.
  \item \textbf{Change control:} Contract tests for resource profiles and webhook payloads; semantic versioning on topics.
\end{itemize}

\paragraph{Governance Layer}
HIPAA/GDPR controls, consent + purpose binding, auditing, equity + neurodiversity policies.
\begin{itemize}
  \item \textbf{Consent:} Explicit purpose-of-use, revocation service, data minimization defaults; consent artifacts linked via \texttt{Provenance}.
  \item \textbf{Privacy:} S2H holds re-ID keys in a KMS-sealed vault; education receives de-ID aggregates only.
  \item \textbf{Audit:} Continuous privacy audit stream; pass-rate KPI; policy drift detection with alerting.
  \item \textbf{Equity \& neurodiversity:} Distribution checks; EchoBraid profiles for ASD; culturally responsive content policies.
  \item \textbf{Sovereignty:} Offline bundles, local LAN operation, delayed sync with CRDT merge and community key custody.
\end{itemize}

\subsection{Reference Dataflow}
\begin{verbatim}
Wearables/EHR -> Intrinsica [ingest -> feature store -> risk] -> ULMS + XR (personalized)
        -> S2H protocols -> outcomes -> Intrinsica (retrain + recalibrate)
                    \-> de-ID aggregates -> HEP dashboards + research
\end{verbatim}

  \texttt{reciprocity.delta.v1}, \texttt{pilila.milestone.v1}, \texttt{privacy.audit.event.v1}.

  \item \textbf{Education zone:} De-ID only; k-anonymity thresholds; no small-cell exports.
  \item \textbf{Shared services:} Policy engine, consent registry, ledger; access via DID-bound OAuth2 and mTLS.
\end{itemize}

\section{Roles and Contracts}

\subsection{Intrinsica (Data/AI Plane)}

\paragraph{Mission}
Operate the single data plane: ingest, featureization, risk inference, provenance, and policy decisions.

\paragraph{Responsibilities}
\begin{itemize}
  \item Lakehouse + feature store; hybrid ML (batch + streaming) for risk and engagement models.
  \item FHIR/HL7 IO for clinical interoperability; API gateway for education-facing services.
  \item De-identification (de-ID) on egress to education; consent registry; permissioned ledger for audit and integrity.
  \item Emit cohort trends and privacy-safe aggregates; maintain model/version catalogs and drift monitors.
\end{itemize}

\paragraph{Inputs}
\begin{itemize}
  \item Clinical: FHIR \texttt{Observation}, \texttt{QuestionnaireResponse}, \texttt{Procedure}, \texttt{MedicationStatement}.
  \item Device/Wearables: time-series biometrics (HRV, sleep, activity) via \texttt{Device}/\texttt{Observation}.
  \item ULMS signals: de-ID \texttt{ulms.session.attempted}, CSL state changes, completion events.
  \item Governance: \texttt{Consent}, \texttt{Provenance}, policy versions.
\end{itemize}

\paragraph{Outputs}
\begin{itemize}
  \item \textbf{Risk vectors} per subject/cohort; \textbf{cohort trends} and \textbf{equity diagnostics}.
  \item \textbf{Provenance} records (signed, hashed); \textbf{policy decisions} for gating and personalization.
  \item De-ID event streams for education dashboards.
\end{itemize}

\paragraph{Interfaces (contracts)}
\begin{itemize}
  \item \textbf{FHIR Endpoint} (R4): RESTful create/read/update, Bulk Data export for cohorts.
  \item \textbf{Subscriptions} (rest-hook): \texttt{Observation?category=survey,vital-signs}, \texttt{QuestionnaireResponse}.
  \item \textbf{Risk API} (\texttt{/risk/v1/assess}): POST subject or cohort; returns vector and explanations.
  \item \textbf{Consent API} (\texttt{/consent/v1/register,\ /revoke}): purpose tags, scope, expiry; returns consent id.
  \item \textbf{Event Bus} (append-only, versioned): \texttt{intrinsica.risk.update.v1}, \texttt{privacy.audit.event.v1}.
\end{itemize}

\paragraph{Example: Risk API}
\begin{verbatim}
POST /risk/v1/assess
{
  "subject_pid": "p_9f3…",
  "context": {"cohort": "hs-10", "window": "7d"},
  "features_hint": ["hrv_rmssd", "sleep_eff"]
}
-- Response --
{
  "risk": {"anxiety": 0.62, "sleep": 0.74},
  "explanations": [{"feature":"hrv_rmssd","weight":0.31}],
  "provenance": {"hash":"b64…","policy":"v1.2","model":"sleep-risk-2025.09"}
}
\end{verbatim}

\paragraph{Security \& Governance}
\begin{itemize}
  \item Re-ID keys held only by S2H; Intrinsica processes PHI under BAA; all education egress is de-ID.
  \item All writes purpose-bound, signed, and hashed to ledger; SAR/erasure honored by policy engine.
\end{itemize}

\paragraph{SLOs}
\begin{itemize}
  \item Online inference p50 $<80$\,ms; ingest$\rightarrow$feature materialization p95 $<5$\,min.
  \item Ledger commit $<2$\,s p95; de-ID egress within $<60$\,s of event receipt.
\end{itemize}

\paragraph{Failure Modes \& Backoffs}
\begin{itemize}
  \item Policy cache miss $\rightarrow$ deny-by-default; queue until policy fetch.
  \item Model outage $\rightarrow$ fallback policy; emit \texttt{privacy.audit.event.v1} with \texttt{mode:"degraded"}.
\end{itemize}

\section{Security, Privacy, Governance}

\subsection{Principles}
\begin{enumerate}
  \item \textbf{Purpose-binding}: every write carries explicit purpose-of-use.
  \item \textbf{Least privilege}: scoped, time-bound access tokens; mTLS between zones.
  \item \textbf{De-identification by default}: education zone receives de-ID aggregates only.
  \item \textbf{Traceability}: all events signed, hashed, and anchored to a permissioned ledger.
  \item \textbf{Sovereignty}: offline-first operation with community-held keys and delayed sync.
\end{enumerate}

\subsection{HIPAA/GDPR Alignment}
\begin{itemize}
  \item \textbf{HIPAA}: access controls, audit controls, integrity, transmission security; BAAs in place.
  \item \textbf{GDPR}: lawfulness and purpose limitation; data minimization; SAR/portability; erasure and revocation; DPIA for high-risk pilots.
  \item \textbf{Operationalization}: policy engine enforces purpose tags; processors blocked if policy mismatch.
\end{itemize}

\subsection{Sovereignty}
\begin{itemize}
  \item \textbf{Community keys}: DID-based identities; local custodians rotate keys on schedule.
  \item \textbf{Offline bundles}: content and policy packs signed; CRDT-based merge on reconnect.
  \item \textbf{Delayed sync}: bounded staleness targets; conflict resolution rules favor newest policy version.
\end{itemize}

\subsection{Ledger and Provenance}
\begin{itemize}
  \item \textbf{Model}: permissioned, append-only; entries store \{\texttt{event\_hash}, \texttt{policy\_version}, \texttt{did}, \texttt{ts}\}.
  \item \textbf{Usage}: every FHIR write and ULMS webhook emits a provenance record; ledger stores content-hash, not payload.
  \item \textbf{Assurance}: verifiable chain for audits, incident postmortems, and DPIA evidence.
\end{itemize}

\subsection{Re-identification Boundary}
\begin{itemize}
  \item \textbf{Rule}: re-ID only inside the S2H enclave via KMS-sealed token vault.
  \item \textbf{Dual control}: emergency re-ID requires two approvers (clinician + steward); ledger-logged.

\subsection{De-ID Pipeline (Stream)}
\paragraph{Steps}
\begin{enumerate}
  \item Tokenize \(\mathrm{PID}\rightarrow\mathrm{pseudo\_id}\) using format-preserving encryption (FPE).
  \item Drop direct identifiers; generalize quasi-identifiers to cohort bins (age-band, zipcode prefix).
  \item Attach \texttt{provenance+policy} and purpose; hash payload; emit de-ID topic.
\end{enumerate}

\paragraph{Pseudo-code}
\begin{verbatim}
function deid(event):
  pid = event.subject_pid
  pseudo = FPE.encrypt(pid, key=K_deid)
  payload = strip_identifiers(event)
  payload.subject = {"pseudo_id": pseudo}
  payload.quasi = generalize(payload.quasi)
  env = {policy: P.current, purpose: event.purpose}
  hash = blake3(serialize(payload||env))
  emit(topic=event.topic+".deid", body=payload, provenance={hash, env})
\end{verbatim}

\section{Feedback Loops (Instrumented)}

\subsection{Overview}
Three loops are implemented and measured: \emph{Personalization}, \emph{Translation}, and \emph{R\&D}. Each loop has inputs, decisions, actions, telemetry, KPIs, SLOs, and failure backoffs.

\subsection{Loop A: Personalization}\label{sec:loop-personalization}

\paragraph{Narrative}
Wearables/EHR $\rightarrow$ Intrinsica risk $\rightarrow$ XR micro-interventions in S2H + HEP $\rightarrow$ new signals. CSL rate-limits, Reciprocity balances cohort energy, $\pi$Līla tracks growth.

\paragraph{Inputs}
\begin{itemize}
  \item \textbf{Signals}: HRV, sleep efficiency, activity, PROMs, ULMS attempts (de-ID).
  \item \textbf{Context}: cohort, accommodations, cultural pack, prior $\pi$Līla milestones.
  \item \textbf{Policies}: purpose tags, throttle bounds, inclusion rules.
\end{itemize}

\paragraph{Decisions}
\begin{enumerate}
  \item \textbf{Risk mapping}: $r=\mathrm{RiskModel}(x)$ from feature vector $x$.
  \item \textbf{CSL gating}: compute load $L \in [0,1]$, gate $\in \{\mathrm{open},\mathrm{cooldown},\mathrm{closed}\}$ with hysteresis.
  \item \textbf{Reciprocity balance}: maintain cohort energy $E_t$; rate-limit group asks if out-of-band.
  \item \textbf{$\pi$Līla progression}: select next milestone step and intensity.
\end{enumerate}

\paragraph{CSL Algorithm (hysteresis)}
\begin{verbatim}
Inputs: load_score L, thresholds θ_open<θ_cool<θ_close, cooldown τ
State: gate ∈ {open,cooldown,closed}, t_last
if gate==open    and L>θ_cool  -> gate=cooldown; t_last=now
if gate==cooldown and L<θ_open and now-t_last>τ -> gate=open
if L>θ_close -> gate=closed (human-in-loop); emit alert
\end{verbatim}

\paragraph{Reciprocity Balance}
\[
E_t = \lambda E_{t-1} + \alpha \cdot \text{inflow}_t - \beta \cdot \text{outflow}_t
\]
If $E_t \notin [E_{\min},E_{\max}]$, reduce group intensity by factor $\gamma \in (0,1)$ and schedule restitution tasks. Emit \texttt{reciprocity.delta.v1}.

\paragraph{$\pi$Līla Milestones}
Let path $P$ have ordered milestones $m_1,\dots,m_k$. Advance when criteria met:
\[
\text{advance if } \text{TCI}_P \in [t_{\min}, t_{\max}] \wedge \text{CSL}=\mathrm{open} \wedge \Delta r \ge \delta
\]
Emit \texttt{pilila.milestone.v1}.

\paragraph{Actions}
\begin{itemize}
  \item Personalize module parameters: target cadence, difficulty band, session length.
  \item Switch to low-intensity variant if gate $\ne$ open, preserve continuity.
  \item Notify clinician (S2H) on sustained overload or risk spike.
\end{itemize}

\paragraph{Telemetry (events)}
\begin{itemize}
  \item \texttt{intrinsica.risk.update.v1} (per subject/cohort).
  \item \texttt{csl.state.change.v1} (gate transitions).
  \item \texttt{ulms.session.attempted.v1} (de-ID attempts).
  \item \texttt{reciprocity.delta.v1}, \texttt{pilila.milestone.v1}, \texttt{privacy.audit.event.v1}.
\end{itemize}

\paragraph{KPIs}
\begin{itemize}
  \item Risk-score delta (30d): $\overline{\Delta z(r)}$.
  \item Overload rate: $\frac{\#\{\text{gate}\neq\mathrm{open}\}}{\#\text{sessions}}$.
  \item Time-to-next-milestone: median days per $\pi$Līla step.
\end{itemize}

\paragraph{SLOs}
\begin{itemize}
  \item Personalization p50 $<100$\,ms; CSL decision p50 $<50$\,ms.
  \item Event end-to-end (device$\rightarrow$dashboard) p95 $<5$\,min cloud; $<24$\,h sovereign.
\end{itemize}

\paragraph{Failure Modes \& Backoffs}
\begin{itemize}
  \item Model unavailable $\rightarrow$ fallback policy band; log degraded mode.
  \item Policy fetch fail $\rightarrow$ deny-by-default; queue with retry + jitter.
  \item High overload streak $\rightarrow$ human-in-loop, shorter sessions, content swap.
\end{itemize}

\paragraph{CSL Gating}

\begin{verbatim}

if gate==cooldown and L<θ_o -> gate=open

\end{verbatim}

\section{KPIs and Formulas}

\subsection{Notation and Windows}
Let learners be indexed by $i$, sessions by $s$, days by $t$, cohorts by $c$. Rolling window $W$ defaults to 30 days unless noted. Aggregation uses either macro average (per-learner mean) or micro average (global ratio). Denominators must be $>0$.

\subsection{Core KPIs}

\paragraph{XR engagement}
Per learner: $E_i=\dfrac{\text{completed\_sessions}_i}{\text{assigned\_sessions}_i}$.
\[
E_{\text{macro}}=\frac{1}{|L|}\sum_{i\in L}E_i,\qquad
E_{\text{micro}}=\frac{\sum_i \text{completed\_sessions}_i}{\sum_i \text{assigned\_sessions}_i}
\]

\paragraph{Risk-score delta}
Let $r_{i,t}$ be a risk score and $\mu,\sigma$ be baseline mean and std from a pre period $B$. Define $z_{i,t}=\dfrac{r_{i,t}-\mu}{\sigma}$. The 30-day change for learner $i$ is
\[
\Delta z_{i}^{30}=\frac{1}{|W|}\sum_{t\in W}\left(z_{i,t}-z_{i,t-30}\right)
\]
Cohort aggregate: $\Delta z_{c}^{30}=\frac{1}{|L_c|}\sum_{i\in L_c}\Delta z_{i}^{30}$.

\paragraph{Adherence}
Dose adherence: $A^{\text{dose}}_i=\dfrac{\text{on\_time\_doses}_i}{\text{scheduled\_doses}_i}$.
Plan-day adherence: $A^{\text{plan}}_i=\dfrac{\text{plan\_days\_completed}_i}{\text{plan\_days\_assigned}_i}$.

\paragraph{ULMS completion}
$C_i=\dfrac{\text{completed\_micro\_lessons}_i}{\text{assigned\_micro\_lessons}_i}$.
Macro or micro aggregation as in engagement.

\paragraph{Equity gap closed}
Let $g$ index groups or use distributional bands. Default distributional gap: $G_{\text{now}}=P90-P10$ for the outcome of interest. With baseline $G_0$,
\[
\text{GapClosed}=\frac{G_0-G_{\text{now}}}{\max(G_0,\epsilon)}
\]
Group-wise variant: $\text{GapClosed}=\dfrac{(\bar{y}_{\max,0}-\bar{y}_{\min,0})-(\bar{y}_{\max}-\bar{y}_{\min})}{\max(\bar{y}_{\max,0}-\bar{y}_{\min,0},\epsilon)}$.

\paragraph{Privacy audit pass-rate}
\[
P_{\text{privacy}}=\frac{\text{compliant\_events}}{\text{total\_auditable\_events}}
\]
Constraint hits, policy drift, and denied writes are noncompliant.

\subsection{Learning Process Metrics}

\paragraph{PER richness}
Let $k$ be artifact types and $w_k$ weights. For learner $i$ in window $W$,
\[
\text{PER}_{i}=\frac{\sum_{a\in W}\sum_{k} w_k \cdot \mathbf{1}\{\text{artifact}_{a,k}\}}{\sum_{a\in W}1}
\]

\paragraph{MLEM}
Gain units $G_i$ and effort units $E_i$ over $W$,
\[
\text{MLEM}_{i}=\frac{G_i}{\max(1,E_i)}\quad\text{with}\quad G_i=\Delta \text{TCI mastery or assessment delta},\ E_i=\text{time on task}+\text{hints}
\]

\paragraph{TCI state}
Time on concept $T_{i,c}$ vs adaptive band $[t_{\min},t_{\max}]$,
\[
\text{state}_{i,c}=\begin{cases}
\text{rising} & T_{i,c}\in[t_{\min},t_{\max}] \\
\text{overload} & T_{i,c}>t_{\max} \\
\text{underload} & T_{i,c}<t_{\min}
\end{cases}
\]
Stability is the share of sessions with nonoverload states.

\subsection{Aggregation and Robustness}
\begin{itemize}
  \item Winsorize outliers at 1st and 99th percentiles before cohort means.
  \item Enforce $k$-anonymity on subgroup reports; suppress cells with $n<k$.
  \item Prefer micro averages when assignment counts vary widely.
\end{itemize}

\subsection{Service Levels for Analytics}
\begin{itemize}
  \item Risk delta materialization p95 $<5$ min cloud, $<24$ h sovereign.
  \item Dashboard freshness p95 $<10$ min for cloud cohorts.
\end{itemize}

\subsection{SQL and Pseudocode}

\paragraph{Risk-score delta (30d)}
\begin{verbatim}
-- risk_daily: columns [subject_id, cohort, date, risk_z_today, risk_z_30d_ago]
SELECT cohort, AVG(risk_z_today - risk_z_30d_ago) AS delta30
FROM risk_daily
WHERE date BETWEEN :start AND :end
GROUP BY cohort;
\end{verbatim}

\paragraph{Equity gap closed}
\begin{verbatim}
-- outcomes_dist: columns [date, metric, p10, p90, baseline_gap]
SELECT (baseline_gap - (p90 - p10)) / NULLIF(baseline_gap,0) AS gap_closed
FROM outcomes_dist
WHERE metric = :metric AND date BETWEEN :start AND :end;
\end{verbatim}

\paragraph{XR engagement (micro average)}
\begin{verbatim}
SELECT SUM(completed_sessions)::float / NULLIF(SUM(assigned_sessions),0) AS engagement_micro
FROM xr_assignments
WHERE date BETWEEN :start AND :end;
\end{verbatim}

\paragraph{ULMS completion (macro average)}
\begin{verbatim}
WITH per_learner AS (
  SELECT learner_id,
         SUM(completed_micro_lessons)::float / NULLIF(SUM(assigned_micro_lessons),0) AS comp_i
  FROM ulms_lessons
  WHERE date BETWEEN :start AND :end
  GROUP BY learner_id
)
SELECT AVG(comp_i) AS completion_macro FROM per_learner;
\end{verbatim}

\paragraph{Privacy audit pass-rate}
\begin{verbatim}
SELECT SUM(CASE WHEN compliant THEN 1 ELSE 0 END)::float
       / NULLIF(COUNT(*),0) AS privacy_pass_rate
FROM privacy_audit_events
WHERE ts BETWEEN :start AND :end;
\end{verbatim}

\paragraph{TCI stability}
\begin{verbatim}
WITH states AS (
  SELECT learner_id,
         SUM(CASE WHEN state IN ('rising','underload') THEN 1 ELSE 0 END)::float
         / NULLIF(COUNT(*),0) AS stability
  FROM tci_events
  WHERE ts BETWEEN :start AND :end
  GROUP BY learner_id
)
SELECT AVG(stability) AS tci_stability_macro FROM states;
\end{verbatim}

\subsection{R1 — Re-identification}
\paragraph{Threats}
Linkage across public datasets, small-cell releases, quasi-identifier joins, membership inference on aggregates.

\paragraph{Prevention Controls}
\begin{itemize}
  \item \textbf{k-anonymity}: enforce $k \ge 10$ on any cohort export; suppress if $n<k$.
  \item \textbf{l-diversity / t-closeness}: for sensitive attributes ensure $l \ge 2$ and $D_{TV}(\text{group},\text{pop}) \le t$.
  \item \textbf{Noise for public stats}: Laplace mechanism with $\epsilon \le 1.0$ on open dashboards; integer rounding and top/bottom coding.
  \item \textbf{Thresholding}: \emph{no release} for slices with $n<k$ or $p\in\{0,1\}$.
  \item \textbf{Zone isolation}: re-ID keys only in S2H enclave; education receives de-ID aggregates.
  \item \textbf{Minimization}: drop direct identifiers; generalize quasi-identifiers (age-bands, geo prefixes).
\end{itemize}

\paragraph{Detection}
\begin{itemize}
  \item Uniqueness scans on exports; linkage attempt heuristics; query patterns that narrow to $n<k$.
  \item \texttt{privacy.audit.event.v1} on policy mismatch, small-cell detection, or DP budget exhaustion.
\end{itemize}

\paragraph{Response}
\begin{enumerate}
  \item Embargo offending export; rotate de-ID keys; raise $k$ and increase noise.
  \item Ledger-stamped postmortem; notify stewards; update DPIA if applicable.
\end{enumerate}

\paragraph{Metrics}
\[
\text{small\_cell\_exposures} = \sum_{\text{exports}} \mathbf{1}\{n<k\},\quad
\text{dp\_budget\_spent} = \sum \epsilon_{\text{queries}} \le \epsilon_{\max}
\]

\paragraph{De-ID Guard (pseudo)}
\begin{verbatim}
if cohort_size < k or p in {0,1}: deny("small-cell")
value := value + Laplace(1/epsilon); round(); publish()
\end{verbatim}

\subsection{R2 — Content Drift}
\paragraph{Threats}
Unreviewed XR updates, A/B outside guardrails, content-policy mismatch during audits.

\paragraph{Prevention Controls}
\begin{itemize}
  \item \textbf{Versioning}: semver for modules; immutable, signed bundles.
  \item \textbf{Guardrails}: pre-registered A/B ranges; CSL limits enforced at runtime.
  \item \textbf{Change control}: two-person review; freeze windows during audits.
\end{itemize}

\paragraph{Detection}
\begin{itemize}
  \item Hash mismatch alarms; outcome skew monitors (PSI/KL) vs. baseline.
  \item \texttt{privacy.audit.event.v1} for policy-version drift.
\end{itemize}

\paragraph{Response}
\begin{enumerate}
  \item Auto-rollback to last signed bundle; disable experiment flags.
  \item Open incident; root-cause within 72\,h; re-release under change ticket.
\end{enumerate}

\paragraph{Metrics}
\[
\text{drift\_alerts} \downarrow,\quad \text{rollback\_MTTR} < 2\text{ h},\quad
\text{policy\_mismatch} = 0
\]

\subsection{R3 — Load/Overwhelm}
\paragraph{Threats}
Cognitive overload, extended sessions, network retries causing rapid-fire prompts.

\paragraph{Prevention Controls}
\begin{itemize}
  \item \textbf{CSL gating}: hysteresis thresholds $(\theta_o,\theta_c,\theta_x)$; cooldowns.
  \item \textbf{Outage-safe backoffs}: exponential with jitter; never repeat high-load tasks on retry.
  \item \textbf{Session caps}: max block lengths and daily limits per skin.
\end{itemize}

\paragraph{CSL Backoff (algorithm)}
\begin{verbatim}
if L > θ_c -> gate=cooldown for τ; next_variant=intensity * γ (γ<1)
if L > θ_x -> gate=closed; alert human-in-loop; suspend high-intensity tracks
\end{verbatim}

\paragraph{Detection}
\begin{itemize}
  \item Overload streaks per learner; spike in \texttt{csl.state.change} to non-open.
  \item Dropout uplift post-intensity increase.
\end{itemize}

\paragraph{Response}
\begin{enumerate}
  \item Switch to low-intensity variants; shorten blocks; increase spacing.
  \item Clinician/educator review for repeated overload $(\ge 3$ events/week$)$.
\end{enumerate}

\paragraph{Metrics}
\[
\text{overload\_rate}=\frac{\#\{\text{gate}\neq\text{open}\}}{\#\text{sessions}} \le 0.10,\quad
\text{cooldown\_median}\le 180\text{ s}
\]

\subsection{R4 — Sovereign Sites}
\paragraph{Threats}
Delayed sync, divergent edits, key loss, side-loaded content.

\paragraph{Prevention Controls}
\begin{itemize}
  \item \textbf{CRDT data types}: counters, sets, maps with per-field merge.
  \item \textbf{Vector clocks}: detect concurrency; policy versions resolve ties (highest wins).
  \item \textbf{Local key custody}: community HSM; multi-sig for key ops; paper escrow.
  \item \textbf{Signed firebooks}: content/policy bundles with checksums.
\end{itemize}

\paragraph{Conflict Resolution (CRDT sketch)}
\begin{verbatim}
merge(a,b):
  for field in state:
    if causal(a,f) > causal(b,f): out[f]=a[f]
    else if ... : out[f]=b[f]
    if tie: out[f]=policy_version_max(a,b).apply(f)
\end{verbatim}

\paragraph{Detection}
\begin{itemize}
  \item Sync lag monitors; conflict counts; signature verification failures.
\end{itemize}

\paragraph{Response}
\begin{enumerate}
  \item Queue conflicts to steward console; apply deterministic merge; re-stamp ledger.
  \item Key-loss: rotate community keys; re-issue device certs; invalidate old bundles.
\end{enumerate}

\paragraph{Metrics}
\[
\text{sync\_lag\_p95} < 24\text{ h},\quad \text{conflicts\_resolved\_p95} < 4\text{ h},\quad
\text{sig\_failures}=0
\]

  \item \textbf{Testing}: synthetic linkage attacks; chaos on event bus; A/B guardrail fuzzing; sovereign offline drills.

\subsection*{Nexus–Native Constructs}
\begin{description}
  \item[PER] Process-evidence record of attempts and artifacts. \emph{Origin: M-Education Health Nexus pedagogy.}
  \item[MLEM] Meta-learning efficiency metric; learning gain per unit effort. \emph{Origin: M-Education Health Nexus analytics.}
  \item[TCI] Time-on-concept index with mastery state bands. \emph{Origin: M-Education Health Nexus analytics.}
  \item[CSL] Cognitive Strain Limiter; gates intensity with hysteresis and cooldown. \emph{Origin: M-Education Health Nexus safety.}
  \item[Reciprocity] Cohort energy balance over a rolling window; throttles group asks. \emph{Origin: Nexus cohort dynamics.}
  \item[$\pi$Līla] Progression metaphor; milestone tracking along growth paths. \emph{Origin: Nexus learning design.}
  \item[ULMS] Universal Learning Management Services layer and schema. \emph{Origin: Nexus platform architecture.}
  \item[XR MaaS] Catalog of XR micro-modules delivered as a service, dual-skinned. \emph{Origin: Nexus content system.}
  \item[Risk vector] Model output per subject/cohort for sleep, stress, adherence, etc. \emph{Origin: Intrinsica analytics.}
  \item[G$\leftrightarrow$N$\leftrightarrow$Q] Frame-toggling task design: Graphical, Numerical, Qualitative. \emph{Origin: Q-Education framework (Nexus).}
  \item[EchoBraid] Sensory-rhythm accommodations for ASD; pacing, envelopes, turn-taking. \emph{Origin: Nexus ASD protocol.}
  \item[Firebook] Signed offline bundle with assets, policies, and tests for sovereign sites. \emph{Origin: Nexus sovereignty ops.}
\end{description}

\subsection*{Citizen Gardens Overlay}

  \item[Citizen Gardens (CG)] Community-embedded governance and ethics layer. \emph{Origin: Citizen Gardens social design.}
  \item[ELM] Economy League Multiplicity; social multiplicity and reciprocity ethic. \emph{Origin: Citizen Gardens framework.}
  \item[Ethical Vital Signs] Dashboard set for engagement, reciprocity, CSL, equity, privacy. \emph{Origin: CG\,\&\,Nexus joint instrumentation.}
\end{description}

\subsection*{Clinical \& Education Interop}
\begin{description}
  \item[FHIR R4] HL7 clinical resource model and REST API. \emph{Origin: HL7.}
  \item[HL7 v2] Event-driven clinical messaging (e.g., ADT). \emph{Origin: HL7.}
  \item[Provenance] FHIR resource capturing signer, hash, policy version. \emph{Origin: HL7 FHIR.}
  \item[Consent] FHIR resource for purpose, scope, revocation. \emph{Origin: HL7 FHIR.}
  \item[ResearchStudy/Subject] FHIR linkage of cohorts to pilots. \emph{Origin: HL7 FHIR.}

\subsection*{Identity, Security, Privacy}
\begin{description}
  \item[DID] Decentralized Identifier used for signing and audit. \emph{Origin: W3C.}
  \item[OAuth2] Service authentication via client-credentials. \emph{Origin: IETF.}
  \item[JWT] Bearer token format; DID-bound claims. \emph{Origin: IETF.}
  \item[mTLS] Mutual TLS between zones. \emph{Origin: IETF/TLS.}
  \item[ABAC] Attribute-based access control (purpose, role, zone). \emph{Origin: NIST SP 800-162.}
  \item[KMS/HSM] Managed key service and hardware security module. \emph{Origin: NIST/FIPS 140-3 practices.}
  \item[FPE] Format-preserving encryption for PID$\rightarrow$pseudo\_id. \emph{Origin: NIST FF3-1.}
  \item[PHI/PII] Protected Health/Personally Identifiable Information. \emph{Origin: HIPAA/Privacy law.}
  \item[HIPAA] US health privacy/security rules. \emph{Origin: HHS.}
  \item[GDPR/DPIA] EU data protection and impact assessment. \emph{Origin: EU GDPR.}
  \item[FERPA] US student record protections. \emph{Origin: US Dept. of Education.}
  \item[k-anonymity] Release threshold to prevent singling out. \emph{Origin: Sweeney (privacy research).}
  \item[$\ell$-diversity/$t$-closeness] Attribute diversity and distribution closeness for safe releases. \emph{Origin: Machanavajjhala et\,al.; Li et\,al.}
