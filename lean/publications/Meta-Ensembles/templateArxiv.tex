\begin{abstract}
We formalize meta-ensembles as discrete-time recursions that combine a finite family of operators via an adaptive gate and a convex aggregator. We give two regimes with rigorous guarantees: a contractive regime on Banach spaces and an averaged-operator regime on Hilbert spaces leveraging Krasnosel'skii--Mann iteration. The framework supports optional composition via embed--glue--retract while preserving stability. We include ready-to-run code snippets and a benchmark plan.
\end{abstract}

\section{Setting and Notation}
Let $(X,\norm{\cdot})$ be a real Banach space, time $t\in\N$, and an index set $\mathcal P=\{p_1,\dots,p_m\}$ with $m\ge1$. A state is $x_t\in X$. An operator family is $\{\Pi_p\}_{p\in\mathcal P}$ with $\Pi_p:X\to X$. A weight vector is $\alpha(t)=(\alpha_p(t))_{p\in\mathcal P}\in\simplex^{m-1}:=\{a\in\R_+^m: \sum_p a_p=1\}$. Optional exogenous terms: drift $B_t\in X$ and perturbation $N_t\in X$.

\section{Assumptions and Update Rule}
\begin{assumption}[Lipschitz operators]\label{ass:lipschitz}
Each $\Pi_p$ is Lipschitz with constant $L_p$ and $\sup_{p}L_p\le\lambda<1$.
\end{assumption}
\begin{assumption}[Convex aggregator]\label{ass:mu}
$\mu(\alpha,z_1,\dots,z_m)=\sum_{p}\alpha_p z_p$ for $(\alpha,z)\in \simplex^{m-1}\times X^m$.
\end{assumption}
\begin{assumption}[Gate]\label{ass:gate}
$\alpha(t)=\gate_{k-1}(x_t)\in \simplex^{m-1}$; $\gate_{k-1}$ is measurable in $x_t$.
\end{assumption}
\begin{assumption}[Errors]\label{ass:errors}
$\sum_t\norm{B_t}<\infty$ and $\sup_t\E\norm{N_t}<\infty$.
\end{assumption}

Define the time-$t$ map
\begin{equation}
F_t(x):=\mu\big(\gate_{k-1}(x),\{\Pi_p(x)\}_{p\in\mathcal P}\big)+B_t+N_t = \sum_{p}\alpha_p(t)\,\Pi_p(x)+B_t+N_t.
\end{equation}
The recursion is $x_{t+1}=F_t(x_t)$.

\paragraph{Optional composition.} Let $\iota:X\hookrightarrow Y$ and $R:Y\to X$ be bounded linear maps with $R\circ\iota=\Id_X$. Let $G:Y^q\to Y$ be nonexpansive. For $q$ ensembles, set $y^{(j)}=\iota\big(F^{(j)}_t(x)\big)$ and $x_{t+1}=R\,G(y^{(1)},\dots,y^{(q)})$.

\section{Basic Properties in the Contractive Regime}
\begin{lemma}[Mixture nonexpansiveness]\label{lem:mixture}
Under \cref{ass:lipschitz,ass:mu,ass:gate}, $\norm{\sum_p \alpha_p(t)\,\Pi_p(x)-\sum_p \alpha_p(t)\,\Pi_p(y)}\le \lambda\,\norm{x-y}$.
\end{lemma}
\begin{proposition}[Contraction with bounded noise]\label{prop:contraction-noise}
Assume \cref{ass:lipschitz,ass:errors}. With $\varepsilon_t:=\norm{B_t}+\E\norm{N_t}$ and any fixed point $x^*$ of $x\mapsto\sum_p\alpha_p(t)\Pi_p(x)$,
\[
\E\,\norm{x_{t+1}-x^*}\le \lambda\,\E\,\norm{x_t-x^*}+\varepsilon_t,
\]
so if $\sup_t\varepsilon_t\le\varepsilon$ then $\limsup_{t\to\infty}\E\,\norm{x_t-x^*}\le \varepsilon/(1-\lambda)$.
\end{proposition}

\section{Averaged-Operator Variant (Hilbert Setting)}
We now work on a real Hilbert space $(X,\inner{\cdot}{\cdot})$.
\begin{assumption}[Averaged operators]\label{ass:averaged}
Each $\Pi_p$ is $\beta_p$-averaged, i.e., $\Pi_p=(1-\beta_p)\Id+\beta_p N_p$ with $N_p$ nonexpansive and $\beta_p\in(0,1]$ \cite[Sec.~4.2]{BauschkeCombettes2017}.
\end{assumption}
\begin{assumption}[Common fixed points]\label{ass:fix}
Let $T_t(x):=\sum_p \alpha_p(t)\,\Pi_p(x)$. Assume $\bigcap_{t\ge0}\mathrm{Fix}(T_t)\neq\varnothing$.
\end{assumption}
\begin{lemma}[Averagedness preserved by convex mixing]\label{lem:avg-mix}
If \cref{ass:averaged} holds then for any $\alpha\in\simplex^{m-1}$, $T_\alpha(x):=\sum_p\alpha_p\Pi_p(x)$ is $\beta$-averaged with $\beta=\sum_p\alpha_p\beta_p$ \cite[Prop.~4.32]{BauschkeCombettes2017}.
\end{lemma}
\begin{theorem}[Krasnosel'skii--Mann with errors]\label{thm:km}
Assume \cref{ass:averaged,ass:gate,ass:errors,ass:fix}. Consider the relaxed update
\begin{equation}
\label{eq:relaxed}
x_{t+1}=(1-\gamma)\,x_t+\gamma\,T_t(x_t)+B_t+N_t,\qquad \gamma\in(0,1).
\end{equation}
Suppose the family $\{T_t\}$ is finite and each map appears infinitely often. Then $(x_t)$ is Fej\'er monotone with respect to $\bigcap_t\mathrm{Fix}(T_t)$ and converges weakly to a point $x^\star$ in this set. If, in addition, one active operator is a contraction or the family is demiregular at $x^\star$, convergence is strong. \cite[Thm.~5.14, Cor.~5.15]{BauschkeCombettes2017}; see also \cite{Mann1953,CombettesPesquet2015}.
\end{theorem}
\begin{corollary}[Firmly nonexpansive case]
If all $\Pi_p$ are firmly nonexpansive ($\beta_p=1/2$), then under the hypotheses of \cref{thm:km} the relaxed iteration \eqref{eq:relaxed} converges; strong convergence holds under any of the standard strengthening conditions (e.g., viscosity/Halpern variants \cite[Sec.~5.2]{BauschkeCombettes2017}).
\end{corollary}

\paragraph{Remark on gating.} Measurable, state-dependent $\alpha(t)\in\simplex^{m-1}$ does not change nonexpansiveness or averagedness; no Lipschitz inflation arises from the gate itself.

\section{Operator Library with Certified Parameters}
Examples with computable constants:
\begin{itemize}[leftmargin=*]
  \item \textbf{Linear:} $x\mapsto Ax+b$, $\lVert A\rVert\le \lambda<1$ (contraction).
  \item \textbf{Gradient step:} $x\mapsto x-\eta\nabla f(x)$, $f$ is $L$-smooth, $\eta\in(0,2/L)$ $\Rightarrow$ averaged with $\beta=\eta L/2$ \cite[Ex.~4.38]{BauschkeCombettes2017}.
  \item \textbf{Proximal:} $x\mapsto \mathrm{prox}_{\eta g}(x)$ is firmly nonexpansive \cite{ParikhBoyd2014}.
  \item \textbf{Contractive denoiser:} $x\mapsto D(x)$ with $\mathrm{Lip}(D)<1$.
\end{itemize}

\section{Gate Design}
Let $S:X\to\R^r$ be a micro-statistic and $g_\theta: \R^r\times\R_+\to \simplex^{m-1}$ a gate; set $\alpha(t)=g_\theta\big(S(x_t),\tau_k\big)$. Examples: softmax at fixed temperature; Top-$K$ followed by projection onto the simplex. Theory requires only measurability and range in $\simplex^{m-1}$.

\section{Algorithmic Template}
\begin{lstlisting}[style=code,language=Python,caption={Core Meta-Ensemble Loop (NumPy/PyTorch-like pseudocode)}]
import numpy as np

# operators: list of callables Pi[p](x)
# gate: returns simplex weights alpha(x) in R^m
# B_t: optional drift sequence; set to zero for theory runs

def meta_ensemble(x0, operators, gate, T, B=None, gamma=1.0):
    x = x0.copy()
    m = len(operators)
    if B is None:
        B = [0 for _ in range(T)]
    for t in range(T):
        alpha = gate(x)                 # in Delta^{m-1}
        z = [op(x) for op in operators]
        Fx = sum(alpha[p] * z[p] for p in range(m)) + B[t]
        x = (1 - gamma) * x + gamma * Fx
    return x
\end{lstlisting}

\section{Evaluation Protocol}
One dataset per domain; metrics: AUROC or macro-F1. Baselines: bagging, XGBoost, stacking, MoE with softmax gate. Ablations: uniform $\alpha$; single operator; remove glue; softmax$\leftrightarrow$Top-$K$; temperature sweep. Use 5 seeds and pre-register operators, metrics, and thresholds.

\section{Limitations}
Convergence speed depends on contraction/averaged constants and perturbations. Nonconvex glue or nonlinear aggregators may break guarantees. Statistical superiority over baselines is empirical and currently unproven.
