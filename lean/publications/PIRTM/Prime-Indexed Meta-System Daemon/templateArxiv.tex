\section{Context and Novelty}
Traditional self-modifying systems (e.g., autonomic computing, evolutionary algorithms) operate on a detect-modify-evaluate cycle where safety and governance evaluations occur \textit{after} the system state has mutated. The Multiplicity PMD inverts this paradigm. In the PMD, the system proposes an upgrade, compiles it through the Prime-Indexed Recursive Transformation Machine (PIRTM), executes it in a simulated digital twin, generates a Zero-Knowledge (ZK) proof of the execution, and submits the proof and state delta to the Phase Mirror policy gate. 

If and only if the Phase Mirror gate returns a \texttt{PASS}, the state transition is committed to the live runtime and recorded to a cryptographic ledger. The ledger therefore does not record arbitrary system history; it acts as a normative record of permitted state transitions.

\begin{enumerate}

\section{Contractivity and Circuit Depth Coupling}
The inner recurrence is defined as \(X_{t+1} = P(\Xi X_t + \Lambda T(X_t) + G_t)\). The system is bounded by a contractivity margin \(\varepsilon\).
The number of steps \(n\) required to converge within \(\delta\) from initial state norm \(\|X_0\|\) is:
\begin{equation}
n \geq \frac{\log(\|X_0\|/\delta)}{\varepsilon}
\end{equation}
Because the ZK Groth16 circuit requires a fixed compilation depth \(N_{\text{circuit}}\) (e.g., 128), we establish a firm lower bound on \(\varepsilon\):
\begin{equation}
\varepsilon_{\min}^{\text{safe}} = \frac{\log(\|X_0\|/\delta)}{N_{\text{circuit}}}
\end{equation}
For \(N_{\text{circuit}} = 128\) and \(\delta = 0.01\), \(\varepsilon_{\min}^{\text{safe}} \approx 0.036\). Adjusting \(\varepsilon\) below this floor would result in an execution trace longer than the ZK circuit can verify, causing a silent soundness failure.

\section{Resolving the Self-Referential Hash Impossibility}
A configuration file cannot contain its own cryptographic hash. To anchor trust without self-reference, we construct a Merkle tree of all immutable files. The root is committed to a genesis ledger entry before the runtime starts.

\subsection{Code Snippet: Atomic Governance Bootstrap}

import hashlib, json, math, time
from pathlib import Path
import sqlite3

IMMUTABLE_FILES = [
    "contracts/system_invariants.yaml",
    "contracts/shared/constants.py",
    "policy/phase_mirror/__init__.py",
    "rollback/kill_switch.py"
]

def merkle_root(hashes: list[str]) -> str:
    layer = [hashlib.sha256(h.encode()).digest() for h in hashes]
    while len(layer) > 1:
        if len(layer) % 2 == 1: layer.append(layer[-1])
        layer = [hashlib.sha256(layer[i] + layer[i+1]).digest() for i in range(0, len(layer), 2)]
    return layer[0].hex()

# 1. Pre-compute deterministic Cantor ID
p_n, q_k = 2, 1009
tx_id = (p_n + q_k) * (p_n + q_k + 1) // 2 + q_k # 512575

# 2. Write tx_id to constants.py BEFORE hashing
constants_path = Path("contracts/shared/constants.py")
content = constants_path.read_text().replace(
    "GOVERNANCE_MERKLE_ROOT_TX_ID: int = 0",
    f"GOVERNANCE_MERKLE_ROOT_TX_ID: int = {tx_id}"
)
constants_path.write_text(content)

# 3. Compute Merkle root of final files
leaf_hashes = {f: hashlib.sha256(Path(f).read_bytes()).hexdigest() for f in IMMUTABLE_FILES}
root = merkle_root(list(leaf_hashes.values()))

# 4. Commit to genesis ledger entry
conn = sqlite3.connect("tooling/ledger/governance.db")
conn.execute("INSERT INTO governance_ledger VALUES (?,?,?,?,?)", 
             (tx_id, "GOVERNANCE_GENESIS", root, "v0.1.0", int(time.time())))
conn.commit()

\subsection{Recurrence Contractivity and Steady-State Bound}
\label{app:recurrence}

Consider the core recurrence
\begin{equation}
    X_{t+1} = P\bigl(\Xi X_t + \Lambda T(X_t) + G_t\bigr),
\end{equation}
where $P:\mathbb{R}^d\to\mathbb{R}^d$ is a projection ($\norm{P}\le 1$), $\Xi,\Lambda\in\mathbb{R}^{d\times d}$,
$T:\mathbb{R}^d\to\mathbb{R}^d$ is Lipschitz with constant $L_T$, and $\norm{G_t}\le G_{\max}$.

\begin{lemma}[One-step contraction]
    For any $X_t\in\mathbb{R}^d$,
    \begin{equation}
        \norm{X_{t+1}} \le (1-\varepsilon)\norm{X_t} + G_{\max},
    \end{equation}
    where $\varepsilon = 1 - \bigl(\norm{\Xi} + \norm{\Lambda}L_T\bigr)$.
\end{lemma}
\begin{proof}
    Using submultiplicativity and the Lipschitz property,
    \begin{align*}
        \norm{X_{t+1}}
        &\le \norm{P}\Bigl(\norm{\Xi X_t} + \norm{\Lambda T(X_t)} + \norm{G_t}\Bigr) \\
        &\le \norm{\Xi}\norm{X_t} + \norm{\Lambda}\norm{T(X_t)} + G_{\max} \\
        &\le \norm{\Xi}\norm{X_t} + \norm{\Lambda}L_T\norm{X_t} + G_{\max} \\
        &= \bigl(\norm{\Xi} + \norm{\Lambda}L_T\bigr)\norm{X_t} + G_{\max} \\
        &= (1-\varepsilon)\norm{X_t} + G_{\max}.
    \end{align*}
\end{proof}

Iterating the lemma yields the exponential convergence bound.

\begin{theorem}[Steady-state bound and convergence rate]
    Let $\varepsilon>0$. Then for all $t\ge 0$,
    \begin{equation}
        \norm{X_t} \le (1-\varepsilon)^t\norm{X_0} + \frac{G_{\max}}{\varepsilon}\bigl(1-(1-\varepsilon)^t\bigr),
    \end{equation}
    and consequently
    \begin{equation}
        \limsup_{t\to\infty}\norm{X_t} \le \frac{G_{\max}}{\varepsilon}.
    \end{equation}
    Moreover, to achieve $\norm{X_t - X_\infty}\le\delta$ it suffices that
    \begin{equation}
        t \ge \frac{\log(\norm{X_0}/\delta)}{\varepsilon}.
    \end{equation}
\end{theorem}
\begin{proof}
    Unroll the recurrence from Lemma~\ref{lem:onestep}:
    \begin{align*}
        \norm{X_t}
        &\le (1-\varepsilon)^t\norm{X_0} + G_{\max}\sum_{k=0}^{t-1}(1-\varepsilon)^k \\
        &= (1-\varepsilon)^t\norm{X_0} + G_{\max}\frac{1-(1-\varepsilon)^t}{1-(1-\varepsilon)} \\
        &= (1-\varepsilon)^t\norm{X_0} + \frac{G_{\max}}{\varepsilon}\bigl(1-(1-\varepsilon)^t\bigr).
    \end{align*}
    The limit follows as $(1-\varepsilon)^t\to0$. Solving
    $(1-\varepsilon)^t\norm{X_0}\le\delta$ for $t$ gives the claimed bound.
\end{proof}

The circuit depth coupling follows directly from this theorem:
for a fixed-depth ZK circuit that can unroll at most $N_{\text{circuit}}$ recurrence steps,
we require $\frac{\log(\norm{X_0}/\delta)}{\varepsilon}\le N_{\text{circuit}}$,
hence
\begin{equation}
    \varepsilon \ge \varepsilon_{\min}^{\text{safe}} := \frac{\log(\norm{X_0}/\delta)}{N_{\text{circuit}}}.
\end{equation}
With $\norm{X_0}=1$, $\delta=0.01$, $N_{\text{circuit}}=128$ we obtain
$\varepsilon_{\min}^{\text{safe}}\approx 0.036$, as used in \texttt{constants.py}
\cite{code_file:136}.

\subsection{Operator Norm Bounds for the Five-Stage Pipeline}
\label{app:pipeline}

The end-to-end pipeline $\Phi = \phi_5\circ(\phi_4\times\mathrm{id})\circ\phi_3\circ\phi_2\circ\phi_1$
has stages:
\begin{align*}
    \phi_1&:\text{NL}\to\text{PIRTM}_{\text{expr}}, &
    \phi_2&:\text{PIRTM}_{\text{expr}}\to\text{ACE}_{\text{typed}}, &
    \phi_3&:\text{ACE}_{\text{typed}}\to\text{ZK}_{\text{certified}},\\
    \phi_4&:\text{ZK}_{\text{certified}}\times\text{NL}\to\text{Policy}_{\text{gate}}, &
    \phi_5&:\text{Policy}_{\text{gate}}\to\text{Ledger}.
\end{align*}
Their operator norms satisfy:
\begin{align}
    L_{\phi_1} &\le \sup_i r_i\cdot\varepsilon_{\text{grammar}}, \label{eq:L1}\\
    L_{\phi_2} &= 1-\varepsilon, \label{eq:L2}\\
    L_{\phi_3} &\le 1, \label{eq:L3}\\
    L_{\phi_4}(m) &=
    \begin{cases}
        0.90 & m=\text{expression},\\
        0.80 & m=\text{state},
    \end{cases} \label{eq:L4}\\
    L_{\phi_5} &= 1. \label{eq:L5}
\end{align}
Bounds~\eqref{eq:L1}--\eqref{eq:L5} are obtained from the architectural design:
\begin{itemize}
    \item $\phi_1$ contracts by the weighted sum of recurrence coefficients $r_i$ (see grammar constraints).
    \item $\phi_2$ is the recurrence step itself, hence norm $1-\varepsilon$.
    \item $\phi_3$ is norm‑non‑expanding (proof verification does not expand).
    \item $\phi_4$ contracts by factors $0.90$ (expression) or $0.80$ (state) as measured in the tension vocabulary.
    \item $\phi_5$ is the ledger write, which is norm‑preserving.
\end{itemize}
Using submultiplicativity,
\begin{equation}
    L_\Phi = L_{\phi_1}\,L_{\phi_2}\,L_{\phi_3}\,L_{\phi_4}\,L_{\phi_5}
           \le L_{\phi_1}\cdot(1-\varepsilon)\cdot 1\cdot 0.90\cdot 1,
\end{equation}
where the factor $0.90$ corresponds to the expression‑type evaluation (the worst case for contractivity).
Thus $L_\Phi<1$ is guaranteed whenever
\begin{equation}
    L_{\phi_1} < \frac{1}{0.90\,(1-\varepsilon)}.
\end{equation}
For the default $\varepsilon=0.05$ this yields $L_{\phi_1}<1.17$, a bound enforced by the grammar design.

\subsection{Contractivity of the Outer Self‑Modification Map}
\label{app:outer}

Let $\Sigma$ be the configuration space equipped with a metric $d_\Sigma$ that dominates
the sup‑norm on live state and the discrete distance on governance version.
Define the self‑modification map
\begin{equation}
    \mathcal{F}_{\text{gated}}(\sigma) =
    \begin{cases}
        \mathcal{F}(\sigma) & \text{if }\phi_4(\cdots)=\text{PASS}\text{ and }L_\Phi<1,\\[2pt]
        \sigma & \text{otherwise},
    \end{cases}
\end{equation}
where $\mathcal{F}$ denotes the ideal upgrade computation (steps~1--4 of the loop).

\begin{theorem}[Outer loop contraction]
    $\mathcal{F}_{\text{gated}}$ is a contraction on $(\Sigma,d_\Sigma)$ with Lipschitz constant
    $\lambda = \max\bigl\{0,\,L_{\mathcal{F}}\bigr\} < 1$.
\end{theorem}
\begin{proof}
    If the gate returns FAIL, $\mathcal{F}_{\text{gated}}(\sigma)=\sigma$, so
    $d_\Sigma\bigl(\mathcal{F}_{\text{gated}}(\sigma),\mathcal{F}_{\text{gated}}(\sigma')\bigr)=0$
    for any $\sigma,\sigma'$; hence $\lambda_{\text{FAIL}}=0$.
    If the gate returns PASS, by construction we have $L_{\mathcal{F}}<1$ (enforced by the
    watchdog on $L_\Phi$). Thus
    \begin{equation}
        d_\Sigma\bigl(\mathcal{F}_{\text{gated}}(\sigma),\mathcal{F}_{\text{gated}}(\sigma')\bigr)
        = d_\Sigma\bigl(\mathcal{F}(\sigma),\mathcal{F}(\sigma')\bigr)
        \le L_{\mathcal{F}}\,d_\Sigma(\sigma,\sigma').
    \end{equation}
    Taking the worst case over the two branches gives
    $\lambda = \max\{0, L_{\mathcal{F}}\} < 1$.
\end{proof}
By the Banach Fixed‑Point Theorem, $\mathcal{F}_{\text{gated}}$ possesses a unique fixed point
$\sigma^*\in\Sigma$, which is the stable certified configuration toward which the system converges.
