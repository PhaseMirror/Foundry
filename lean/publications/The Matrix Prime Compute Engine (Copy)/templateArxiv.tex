\begin{theorem}[Banach Contractivity of the Meta-Recursive Map (Universal Stable Contractor)]

\[
M(T) = \sum_{p_i \in P_N} \Lambda_m \cdot p_i^\alpha \cdot T_{p_i}^{(m,n)} + F^{(m,n)},
\]
where \(\Lambda_m = \sup_t \rho(\Lambda_m^{\rm op}(t)) \in (0,1)\) is the scalar reduction of the sealed two-layer operator, \(\alpha < -1\), and \(P_N\) is any finite initial segment of the primes. Define the contraction parameter
\[
k = \sum_{p_i \in P_N} \Lambda_m \cdot p_i^\alpha < 1.
\]
Then \(M\) is a strict contraction mapping with Lipschitz constant \(k\), and therefore possesses a unique fixed point \(T_\infty^{(m,n)}\) to which the iteration \(T_{t+1} = M(T_t)\) converges geometrically from any initial tensor \(T_0\):
\[
\|T_t - T_\infty\| \leq k^t \|T_0 - T_\infty\| \to 0 \quad \text{as } t \to \infty.
\]
Moreover, the fixed point admits the explicit closed form
\[
T_\infty^{(m,n)} = \frac{F^{(m,n)}}{1 - k}.
\]
\end{theorem}

\begin{proof}
The space \(\mathcal{T}^{(m,n)}\) is a Banach space (complete normed vector space) under the operator norm inherited from the prime-indexed direct-sum structure \(\mathcal{H}\). The map \(M\) is affine:
\[
M(T) = A(T) + F^{(m,n)},
\]
where the linear part is
\[
A(T) = \sum_{p_i \in P_N} \Lambda_m \cdot p_i^\alpha \cdot T_{p_i}^{(m,n)}.
\]
For any two tensors \(T, S \in \mathcal{T}^{(m,n)}\) we have
\[
\|M(T) - M(S)\| = \|A(T) - A(S)\| = \Bigl\| \sum_{p_i \in P_N} \Lambda_m \cdot p_i^\alpha \cdot (T_{p_i}^{(m,n)} - S_{p_i}^{(m,n)}) \Bigr\| \leq \sum_{p_i \in P_N} \Lambda_m \cdot p_i^\alpha \cdot \|T_{p_i}^{(m,n)} - S_{p_i}^{(m,n)}\| \leq k \|T - S\|,
\]
where the last inequality follows from the definition of \(k\) and the submultiplicativity of the norm (the prime-indexed decomposition is orthogonal with respect to the chosen inner product). Since \(k < 1\), \(M\) is a strict contraction.

By the Banach fixed-point theorem on the complete metric space \(\mathcal{T}^{(m,n)}\), there exists a unique fixed point \(T_\infty\) satisfying \(M(T_\infty) = T_\infty\). Solving the fixed-point equation explicitly yields
\[
T_\infty = A(T_\infty) + F^{(m,n)} \implies T_\infty - A(T_\infty) = F^{(m,n)} \implies (I - A)T_\infty = F^{(m,n)} \implies T_\infty = (I - A)^{-1} F^{(m,n)} = \frac{F^{(m,n)}}{1 - k},
\]
where the Neumann series for \((I - A)^{-1}\) converges absolutely because \(\|A\| \leq k < 1\).

The geometric convergence of the Picard iterates follows directly from the contraction inequality:
\[
\|T_{t+1} - T_\infty\| = \|M(T_t) - M(T_\infty)\| \leq k \|T_t - T_\infty\| \implies \|T_t - T_\infty\| \leq k^t \|T_0 - T_\infty\|.
\]
The rate is independent of the prime truncation \(N\) (so long as \(k < 1\) holds) and of the rank \((m,n)\), making the contractor universal within the multiplicity tensor space.
\end{proof}
