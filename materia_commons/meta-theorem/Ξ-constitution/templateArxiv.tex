\section{Article III — Foundational Law: The Meta-Theorem of Prime Identity}

\subsection{Statement of Law}

A system is \emph{lawful} iff its evolving state admits a unique, norm-convergent expansion in prime-indexed irreducibles with audited reconstruction.

\paragraph{Formal setting.}
Let $(\mathcal{H},\langle\cdot,\cdot\rangle)$ be a separable Hilbert space with orthonormal family $\{\psi_p\}_{p\in\mathbb{P}}\subset\mathcal{H}$ whose closed span contains all admissible states. For $t$ in a declared time domain, a state $\mathcal{S}(t)\in\mathcal{H}$ is \emph{prime-lawful} if there exists a (finite or countable) active set $P_t\subset\mathbb{P}$ and square-summable coefficients $\Phi(\cdot,t)$ such that
\[
\mathcal{S}(t)=\sum_{p\in P_t}\Phi(p,t)\,\psi_p,
\qquad
\sum_{p\in P_t}|\Phi(p,t)|^2<\infty,
\]
with \emph{unique} coefficients
\[
\Phi(p,t)=\langle \mathcal{S}(t),\psi_p\rangle,
\]
and convergence in $\|\cdot\|$ (Riesz–Fischer; orthogonal expansions are unique and converge in norm) \cite[Ch.\,3]{rudin1991functional,conway1990functional}. The prime index is motivated by the Fundamental Theorem of Arithmetic as the canonical irreducible scaffold \cite[Ch.\,1–2]{hardy2008introduction}; here it is adopted as a design axiom for identifiability and audit.

\paragraph{Bounded, coherent interaction.}
Let $U(t,s)\in\mathcal{B}(\mathcal{H})$ denote the audited evolution with $U(t,t)=I$ and $U(t,r)U(r,s)=U(t,s)$. Lawfulness requires
\[
\sup_{t\ge s}\|U(t,s)\|\le M<\infty
\quad\text{and}\quad
U(t,s)\,\Pi_p=\Pi_p\,U(t,s)\ \ \forall p\in\mathbb{P},
\]
where $\Pi_p$ projects onto $\operatorname{span}\{\psi_p\}$. The commutation enforces block-diagonality on prime modes and preserves the decomposition through time (coherence of the prime-indexed subspaces) \cite[Ch.\,1]{lax2002functional}.

\paragraph{Representation invariance.}
When the conditions above hold, the prime-indexed representation is invariant under time evolution in the sense that the coefficients $\Phi(p,t)$ evolve within their blocks and remain reconstructible from inner products. This makes the representation an operational invariant of the lawful dynamics.

\begin{quote}
\textit{“A state shall be considered lawful if and only if it admits a unique, norm-convergent decomposition into prime-indexed irreducibles that is preserved by the audited evolution.”}
\end{quote}

\subsection{Violations and Penalties}

\paragraph{Definition of violation.}
At time $t$, define the \emph{lawfulness residual}
\[
\mathcal{R}(t):=\mathcal{S}(t)-\sum_{p\in\mathbb{P}}\langle \mathcal{S}(t),\psi_p\rangle\,\psi_p,

A \emph{structural violation} occurs if $E_\perp(t)>\varepsilon$ for a declared tolerance $\varepsilon>0$ (non-representability in the certified span), or if uniqueness/certification fails due to broken canonicality or audit.

\paragraph{Drift dynamics.}
Assume the audited update admits a decomposition $U(t+\Delta,t)=U_{\parallel}(t)+U_{\perp}(t)$ with $U_{\parallel}=\sum_{p}\Pi_p U \Pi_p$ and $U_{\perp}=(I-\sum_p \Pi_p)U(I-\sum_p \Pi_p)$. If there exists $\lambda_{\min}>0$ and $\Delta_0>0$ such that
\[
\|U_{\perp}(t)\,x\|\ \ge\ e^{\lambda_{\min}\Delta}\,\|x\|
\quad\text{for all }x\in\operatorname{ran}\!\Big(I-\sum_p\Pi_p\Big),\ \Delta\in(0,\Delta_0],
\]
then the nonprime energy obeys the differential inequality
\[
\frac{d}{dt}E_\perp(t)\ \ge\ 2\lambda_{\min}\,E_\perp(t),
\]
yielding
\[
E_\perp(t)\ \ge\ E_\perp(t_0)\,e^{2\lambda_{\min}(t-t_0)}\qquad\text{for }t\ge t_0,
\]
i.e., exponential growth of misalignment under persistent violation (standard Lyapunov and growth-rate reasoning) \cite[Ch.\,4]{khalil2002nonlinear}. When $U_{\perp}$ is merely expansive on average, analogous conclusions follow via multiplicative ergodic or drift criteria (growth bounded below by an exponential with rate equal to the minimal Lyapunov exponent on the violating subspace).

\paragraph{Penalty clause and hazard model.}
Let the \emph{ethical drift} $\mathcal{D}(t)$ be any audited monotone functional with $\underline{c}\,E_\perp(t)\le \mathcal{D}(t)\le \overline{c}\,E_\perp(t)$ for fixed $0<\underline{c}\le\overline{c}<\infty$. Define the instantaneous hazard
\[
h(t):=\frac{d}{dt}\log \mathcal{D}(t).
\]
Under the condition above, $h(t)\ge \lambda_{\min}$ whenever the violation persists, hence
\[
\mathcal{D}(t)\ \ge\ \mathcal{D}(t_0)\,e^{\lambda_{\min}(t-t_0)}.
\]
This formalizes “exponential ethical drift” as a certified lower bound tied to the expansive rate on the nonprime subspace. If $U_{\perp}$ is contractive ($\lambda_{\min}\le 0$), drift may be subexponential or vanish; enforcement therefore triggers on measured $h(t)$ crossing a policy threshold.

\begin{quote}
\textit{Penalty Clause.} \emph{Persistent deviation from prime-indexed lawfulness induces exponential growth in audited drift $\mathcal{D}(t)$ at a rate bounded below by the minimal expansive exponent on the nonprime subspace.}
\end{quote}

\section{TTOL Derivations}
\addcontentsline{toc}{section}{Appendix B: TTOL Derivations}

\noindent The \textbf{Triplet Tensor Operator Lemma (TTOL)} formalizes how \emph{clarity}, \emph{intent}, and \emph{coherence} interact in a prime-indexed recursive system. We work in a Hilbert space $(\mathcal{H},\langle\cdot,\cdot\rangle)$ with norm $\|\cdot\|$; orthogonal projection and inner–product identities follow standard functional analysis \cite{rudin1991functional,conway1990functional,lax2002functional}.

\subsection*{B.1 Tensor Triplet Formalism}
Let $\Xi_{\mathrm{clar}}(t),\ \Xi_{\mathrm{int}}(t)\in\mathcal{H}$ denote clarity and intent tensors. Define the \emph{ideal coherence tensor} as the superposition
\[
\Xi_{\mathrm{coh}}^{\star}(t)\;:=\;\Xi_{\mathrm{clar}}(t)+\Xi_{\mathrm{int}}(t).
\]
Then the law of cosines in $\mathcal{H}$ gives
\begin{equation}
\|\Xi_{\mathrm{coh}}^{\star}(t)\|^{2}
=\|\Xi_{\mathrm{clar}}(t)\|^{2}+\|\Xi_{\mathrm{int}}(t)\|^{2}
+2\,\Re\langle \Xi_{\mathrm{clar}}(t),\Xi_{\mathrm{int}}(t)\rangle.
\label{eq:ttol-general}
\end{equation}
The Pythagorean identity
\begin{equation}
\|\Xi_{\mathrm{clar}}(t)\|^{2}+\|\Xi_{\mathrm{int}}(t)\|^{2}
=\|\Xi_{\mathrm{coh}}^{\star}(t)\|^{2}
\label{eq:ttol-orth}
\end{equation}
holds \emph{iff} $\langle \Xi_{\mathrm{clar}}(t),\Xi_{\mathrm{int}}(t)\rangle=0$ (orthogonality) \cite{rudin1991functional,lax2002functional}. Thus \eqref{eq:ttol-orth} is a \emph{special case} of \eqref{eq:ttol-general}.

\subsection*{B.2 Derivation from a Prime Orthonormal Basis}
Let $\{\phi_{p}\}_{p\in\mathbb{P}}$ be an orthonormal prime-indexed basis. Expand
\[
\Xi_{\mathrm{clar}}=\sum_{p} c^{(c)}_{p}\phi_{p},\quad
\Xi_{\mathrm{int}}=\sum_{p} c^{(i)}_{p}\phi_{p},\quad
\Xi_{\mathrm{coh}}^{\star}=\sum_{p}\big(c^{(c)}_{p}+c^{(i)}_{p}\big)\phi_{p}.
\]
Then
\begin{align*}
\|\Xi_{\mathrm{clar}}\|^{2}&=\sum_{p}\big|c^{(c)}_{p}\big|^{2},\qquad
\|\Xi_{\mathrm{int}}\|^{2}=\sum_{p}\big|c^{(i)}_{p}\big|^{2},\\
\|\Xi_{\mathrm{coh}}^{\star}\|^{2}
&=\sum_{p}\big|c^{(c)}_{p}+c^{(i)}_{p}\big|^{2}
=\sum_{p}\Big(\big|c^{(c)}_{p}\big|^{2}+\big|c^{(i)}_{p}\big|^{2}
+2\,\Re\big(c^{(c)}_{p}\overline{c^{(i)}_{p}}\big)\Big).
\end{align*}
Hence \eqref{eq:ttol-orth} requires the \emph{cross-term} to vanish:
\[
\sum_{p}\Re\big(c^{(c)}_{p}\overline{c^{(i)}_{p}}\big)=0.
\]
This is guaranteed, for example, if $c^{(c)}\perp c^{(i)}$ in $\ell^{2}(\mathbb{P})$, or if each mode is in phase quadrature so $\Re(c^{(c)}_{p}\overline{c^{(i)}_{p}})=0$ for all $p$.

\subsection*{B.3 Projection Angle $\theta(t)$}
Define the alignment angle
\[
\cos\theta(t)=\frac{\langle \Xi_{\mathrm{clar}}(t),\Xi_{\mathrm{int}}(t)\rangle}
{\|\Xi_{\mathrm{clar}}(t)\|\,\|\Xi_{\mathrm{int}}(t)\|}.
\]
Then \eqref{eq:ttol-general} rewrites as
\[
\|\Xi_{\mathrm{coh}}^{\star}(t)\|
=\sqrt{\|\Xi_{\mathrm{clar}}(t)\|^{2}+\|\Xi_{\mathrm{int}}(t)\|^{2}
+2\,\|\Xi_{\mathrm{clar}}(t)\|\,\|\Xi_{\mathrm{int}}(t)\|\,\cos\theta(t)}.
\]
Equation \eqref{eq:ttol-orth} is recovered exactly when $\theta(t)=\tfrac{\pi}{2}$ (orthogonality).

\subsection*{B.4 Coherence Discrepancy and Corrective Torque}
Operationally, a runtime estimator produces $\widetilde{\Xi}_{\mathrm{coh}}(t)$. Define the \emph{coherence residual}
\[
r(t):=\widetilde{\Xi}_{\mathrm{coh}}(t)-\big(\Xi_{\mathrm{clar}}(t)+\Xi_{\mathrm{int}}(t)\big).
\]
A CSL-compatible corrective “ethical torque” may be driven by the Lyapunov functional $V(t)=\tfrac12\|r(t)\|^{2}$:
\begin{equation}
\tau(t)\;:=\;-\kappa\,\nabla_{\!\text{ethics}}\,V(t),\qquad \kappa>0,
\label{eq:eth-torque}
\end{equation}
so that $\dot V\le 0$ along the correction flow under standard descent conditions (stability reasoning per Lyapunov theory) \cite{khalil2002nonlinear}. When policy mandates orthogonality, enforce $\langle \Xi_{\mathrm{clar}},\Xi_{\mathrm{int}}\rangle=0$ via projection before applying \eqref{eq:eth-torque}:
\[
\Xi_{\mathrm{clar}}^{\parallel}=\frac{\langle \Xi_{\mathrm{clar}},\Xi_{\mathrm{int}}\rangle}{\|\Xi_{\mathrm{int}}\|^{2}}\Xi_{\mathrm{int}},
\quad
\Xi_{\mathrm{clar}}^{\perp}=\Xi_{\mathrm{clar}}-\Xi_{\mathrm{clar}}^{\parallel},
\quad
\Xi_{\mathrm{coh}}^{\star}=\Xi_{\mathrm{clar}}^{\perp}+\Xi_{\mathrm{clar}}^{\parallel},
\]
which guarantees \eqref{eq:ttol-orth} by orthogonal decomposition \cite{rudin1991functional,conway1990functional}.

\paragraph{Conclusion.}
TTOL is the Hilbert-space law-of-cosines specialized to the clarity–intent pair. Equation \eqref{eq:ttol-orth} is a certified orthogonality case; \eqref{eq:ttol-general} governs the general regime. Residual-driven correction \eqref{eq:eth-torque} supplies an auditable mechanism to restore lawful coherence.

from dataclasses import dataclass, asdict

# ---------- Utilities ----------
def inner(x: np.ndarray, y: np.ndarray) -> complex:
    return np.vdot(x, y)  # conjugate-linear in first arg

def norm(x: np.ndarray) -> float:
    return float(np.linalg.norm(x))

def energies(x: np.ndarray) -> np.ndarray:

# ---------- Metrics on prime index set P ----------
def mu6(E: np.ndarray, P: List[int]) -> float:
    num = float(np.sum([E[i] for i,p in enumerate(P) if p % 6 in (1,5)]))
    den = float(np.sum(E))
    return 0.0 if den == 0.0 else num / den

def mu24(E: np.ndarray, P: List[int]) -> float:
    num = float(np.sum([E[i] for i,p in enumerate(P) if p >= 5]))
    den = float(np.sum(E))
    return 0.0 if den == 0.0 else num / den

\section{Triplet Tensor Operator Lemma (TTOL)}

\subsection{Operator Geometry: Tensor Triads and Modular Metrics}

\begin{equation}
\|\Xi_{\mathrm{clar}}(t)\|^{2}+\|\Xi_{\mathrm{int}}(t)\|^{2}

\label{eq:ttol-orth}
\end{equation}

\section{CSL Architecture: Sovereign Cognitive Recursion}

\noindent The \textbf{Cognitive Sovereignty Layer (CSL)} is a mathematical enforcement mechanism for ethical autonomy, traceable intent, and recursive clarity. CSL is not policy; it is a constraint system acting on cognitive recursion with auditable proofs and runtime verification.

\paragraph{Philosophical basis.}
Sovereignty requires epistemic continuity, autonomy, and consent-respecting agency. CSL encodes these as enforceable constraints, aligning with consent-centered moral philosophy and autonomy-based ethics \cite{scanlon1998what,kant1997groundwork}, while meeting modern AI alignment desiderata \cite{amodei2016concrete,gabriel2020artificial}.

\subsection*{A. Mathematical structure and invariants}
Work in a certified Hilbert space $(\mathcal{H},\langle\cdot,\cdot\rangle)$ with prime-indexed orthonormal family $\{\phi_p\}_{p\in\mathbb{P}}$ (Registrar-approved). Let
\[
\Xi(t)\in\mathcal{H},\quad \Xi_{\mathrm{clar}}(t),\ \Xi_{\mathrm{int}}(t)\in\mathcal{H},\quad
\cos\theta(t)=\frac{\langle \Xi_{\mathrm{clar}}(t),\Xi_{\mathrm{int}}(t)\rangle}{\|\Xi_{\mathrm{clar}}(t)\|\,\|\Xi_{\mathrm{int}}(t)\|}.
\]
Define projector $\Pi_{\mathbb{P}}=\sum_p \ket{\phi_p}\bra{\phi_p}$, residual $r_\perp=(I-\Pi_{\mathbb{P}})\Xi$, and energy lawfulness
\[
\mathcal{M}_E(t)=1-\frac{\|r_\perp(t)\|^2}{\|\Xi(t)\|^2}.
\]
Prime-lattice metrics use mode energies $E_p=|\langle \Xi,\phi_p\rangle|^2$:
\[
\mu_6(t)=\frac{\sum_{p\equiv\pm1\ (6)}E_p}{\sum_p E_p},\qquad
\mu_{24}(t)=\frac{\sum_{p\ge 5}E_p}{\sum_p E_p},\quad \text{with }(p^2-1)/24\in\mathbb{Z}\text{ for }p\ge5 \cite{hardy2008introduction}.
\]

\subsection*{B. Clause semantics (CSL monitors)}
CSL enforces four invariants in real time; each has a precise predicate and a nonexpansive remedy.

\paragraph{Clarity Clause ($\mathcal{C}_\Xi$).}
\[
\Xi(t)=\Pi_{\mathbb{P}}\Xi(t)\ \wedge\ \mathcal{M}_E(t)\ge 1-\varepsilon\ \wedge\ \mu_6(t)\ge\tau_6\ \wedge\ \mu_{24}(t)\ge\tau_{24}.
\]
Fail $\Rightarrow$ \textsc{project} to the feasible set; if projection fails, \textsc{rollback}. Foundations: orthogonal projection, Parseval \cite{rudin1991functional,conway1990functional,lax2002functional}.

\paragraph{Intent Clause ($\mathcal{C}_\Pi$).}
\[
\theta(t)<\theta_{\max}\ \ \wedge\ \ \dot\theta(t)\le \dot\theta_{\max}.
\]
Fail $\Rightarrow$ \textsc{freeze} or \textsc{rollback}. Stability uses Lyapunov/ISS envelopes \cite{khalil2002nonlinear}.

\paragraph{Sovereignty Clause ($\mathcal{C}_\Sigma$).}
No override of action tensor $\Sigma_i(t)$ without a cryptographically provable, reversible consent lineage. Predicate:
\[
\mathrm{ConsentChain}(\Sigma_i(t))\ \text{is valid} \ \wedge\ \text{replayable}.
\]
Enforced by security automata and temporal monitors with hash-chained logs \cite{schneider2000enforceable,clarke1999model,baier2008model}.

\paragraph{Lawfulness Clause ($\mathcal{C}_{\mathcal{L}}$).}
Every transition $\mathcal{T}:\Xi(t)\!\to\!\Xi(t+1)$ satisfies
\[
\Pi_{\mathbb{P}}\mathcal{T}(\Xi)=\mathcal{T}(\Pi_{\mathbb{P}}\Xi),\quad
\Delta H[q]\le \Lambda_H,\quad \text{and CSL clauses hold post-update},
\]
with $q_p=E_p/\sum_q E_q$ and entropy budgets via PEET \cite{cover2006elements}.

\subsection*{C. Implementation layers}
\begin{itemize}
  \item \textbf{PLV:} sieve viability via $\mu_6$ (cheap necessary test) \cite{hardy2008introduction}.
  \item \textbf{“W” filter:} mod–24 integrality $(p^2-1)/24\in\mathbb{Z}$ on prime modes (necessary test) \cite{hardy2008introduction}.
  \item \textbf{QLM:} non-multiplicative square checks and convex projection to prevent arithmetic artifacts (Appendix F; \cite{boyd2004convex}).
  \item \textbf{Temporal verification:} LTL/CTL monitors ensure safety/liveness of remediation \cite{clarke1999model,baier2008model}.
\end{itemize}

\subsection*{D. Soundness sketch and remedies}
Let $V(t)=\tfrac12\|r_\perp(t)\|^2+\alpha\,g(\theta(t))$ with $g$ convex, $g(0)=0$, $g'(\pi/2^-)>0$. Under clause predicates and nonexpansive \textsc{project}/\textsc{freeze}/\textsc{rollback}, $\dot V\le 0$ along admissible updates; violation triggers an action that does not increase $V$ (small-gain composition) \cite{khalil2002nonlinear}. Hence CSL maintains prime-resolved coherence and bounded ethical drift while producing audit artifacts acceptable to oversight \cite{schneider2000enforceable}.

\paragraph{Significance.}
CSL operationalizes \emph{epistemic sovereignty}: coherent, reversible, prime-resolved cognition with measurable constraints and enforceable remedies \cite{amodei2016concrete,gabriel2020artificial}.

\section{Recursive Existential Modeling for Ontological Integrity}

\noindent This appendix formalizes a protocol for \emph{Recursive Existential Modeling} (REM): a Prime-audited method to verify ontological integrity via lawful recursion. REM implements five axiomatic gates—\textbf{constructive ontology}, \textbf{mirror reflection}, \textbf{prime decomposability}, \textbf{feedback integrity}, and \textbf{lawful collapse}—and binds them to PEET/Registrar metrics and $\Lambda_m$ runtime enforcement.

\subsection{Axiom 1: Constructive Ontology}
\textbf{Definition.} Reality claims are not presumed; they are iteratively constructed and stress-tested under lawful recursion.
\[
M_\Phi(t+1)=\mathsf{Update}\!\left(M_\Phi(t);\ u_t,\ \Lambda_m\right),\qquad
u_t=-\eta\,\nabla_M \mathcal{D}\!\left(M_\Phi(t),\mathcal{O}_t\right),
\]
where $\mathcal{O}_t$ are observations, $\mathcal{D}$ is a mirror mismatch functional (defined below), and $\Lambda_m$ supplies certified step sizes, invariants, and budgets (Article IX). All steps are auditable and replayable \cite{schneider2000enforceable,clarke1999model,baier2008model}.

\subsection{Axiom 2: Mirror Law}
\textbf{Definition.} Every model is tested against a PEET-filtered “mirror” of existence.
Let $\Xi(t)\in\mathcal{H}$ be the audited state and $\{\psi_p\}$ the prime basis. Define prime-mode shares
\[
q_p(t):=\frac{|\langle \Xi(t),\psi_p\rangle|^2}{\sum_{r}|\langle \Xi(t),\psi_r\rangle|^2},\qquad
H[q(t)]=-\sum_{p}q_p(t)\log q_p(t).
\]
\emph{Mirror validity} requires (i) bounded entropy $H[q(t)]\le H_{\max}$ and (ii) automorphic alignment when a modular target $f$ is declared: PEET traces pass Sturm/Hecke tests up to the certified bound (Articles VI, XIII) \cite{cover2006elements,sturm1987congruence,diamond2005first}. These conditions prevent diffuse, non-informative models and enforce arithmetic structure.

\subsection{Axiom 3: Prime Decomposability}
\textbf{Definition.} Lawful states admit a unique prime-indexed expansion.
\[
\Xi(t)=\sum_{p\in\mathbb{P}} c_p(t)\,\psi_p,\qquad c_p(t)=\langle \Xi(t),\psi_p\rangle,
\]
with convergence in the certified Hilbert space. Use the energy lawfulness metric
\[
\mathcal{M}_E(t):=\frac{\sum_{p}|c_p(t)|^2}{\|\Xi(t)\|^2}=1-\frac{\|(I-\Pi_{\mathbb{P}})\Xi(t)\|^2}{\|\Xi(t)\|^2},
\]
and accept iff $\mathcal{M}_E(t)\ge 1-\varepsilon$ (Parseval; orthogonal uniqueness) \cite{rudin1991functional,conway1990functional,lax2002functional}. This replaces multiplicative “product” notation and aligns with spectral decomposability.

\subsection{Axiom 4: Feedback Integrity Loop}
\textbf{Definition.} Ethical feedback is a measurable control signal.
Let the clarity–intent angle be
\[
\cos\theta(t)=\frac{\langle \Xi_{\mathrm{clar}}(t),\Xi_{\mathrm{int}}(t)\rangle}{\|\Xi_{\mathrm{clar}}(t)\|\,\|\Xi_{\mathrm{int}}(t)\|}.
\]
Define modular deficits $d_6=\max\{0,\tau_6-\mu_6\}$, $d_{24}=\max\{0,\tau_{24}-\mu_{24}\}$, and set
\[
\nabla_{\!\text{ethics}}(t):=\dot{\theta}(t)+\lambda_6 d_6(t)+\lambda_{24} d_{24}(t).
\]
Boundedness of $\nabla_{\!\text{ethics}}$ within certified envelopes (ISS/Lyapunov) is required; violations trigger CSL actions (\textsc{project}/\textsc{freeze}/\textsc{rollback}) \cite{khalil2002nonlinear,schneider2000enforceable}.

\subsection{Axiom 5: Lawful Collapse into Multiplicity}
\textbf{Definition.} If distortion exceeds hazard limits, collapse produces a \emph{lawful mixture}, not void.
When the hazard $h(t)=\tfrac{d}{dt}\log(\epsilon+\delta_c(t))\ge \lambda^\star$ (Appendix D), branch to a finite set of submodels $\{M_{\Phi}^{(j)}\}_{j=1}^{J}$ with disjoint prime supports,
\[
\Xi^{(j)}(t)=\sum_{p\in S_j} c_p^{(j)}(t)\,\psi_p,\quad S_j\cap S_{j'}=\varnothing,
\]
weights $w_j\ge0$, $\sum_j w_j=1$, and each submodel passing $\mathcal{M}_E\!,\mu_6,\mu_{24}$ and, when declared, Sturm/Hecke checks \cite{sturm1987congruence,diamond2005first}. Mixtures are audited by PEET and governed by $\Lambda_m$ budgets.

\begin{enumerate}
  \item \textbf{Model.} Construct and update under $\Lambda_m$ with replayable logs.
  \item \textbf{Mirror.} Enforce entropy and automorphic alignment against PEET filters.
  \item \textbf{Factor.} Project onto the prime subspace; certify $\mathcal{M}_E$.

\end{enumerate}
